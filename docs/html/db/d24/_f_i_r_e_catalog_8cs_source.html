145public sealed class FileInfoMetadataSource : IMetadataSource
146{
151 public string SourceName => "FILEINFO";
152
161 public Dictionary<string, string> ExtractMetadata(string filePath)
162 {
163 var metadata = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
164 try
165 {
166 var fileInfo = new FileInfo(filePath);
167 if (!fileInfo.Exists) return metadata;
168
169 metadata["CreationTime"] = fileInfo.CreationTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
170 metadata["ModificationTime"] = fileInfo.LastWriteTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
171 metadata["AccessTime"] = fileInfo.LastAccessTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
172 }
173 catch { }
174 return metadata;
175 }
176}

236public sealed class ExifToolMetadataSource : IMetadataSource, IDisposable
237{
238 private readonly ExifTool _exifTool;
239 private bool _disposed;
240
245 public string SourceName => "EXIFTOOL";
246
254 public ExifToolMetadataSource()
255 {
256 _exifTool = new ExifTool();
257 }
258
271 public Dictionary<string, string> ExtractMetadata(string filePath)
272 {
273 var metadata = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
274
275 try
276 {
277 if (!File.Exists(filePath))
278 {
279 return metadata;
280 }
281
282 // Execute ExifTool synchronously and get all metadata
283 var result = _exifTool.ExtractAllMetadataAsync(filePath).GetAwaiter().GetResult();
284
285 if (result != null)
286 {
287 foreach (var kvp in result)
288 {
289 if (!string.IsNullOrWhiteSpace(kvp.Key) && kvp.Value != null)
290 {
291 metadata[kvp.Key] = kvp.Value.ToString() ?? string.Empty;
292 }
293 }
294 }
295 }
296 catch
297 {
298 // Metadata extraction failed; return partial or empty result.
299 }
300
301 return metadata;
302 }
303
311 public void Dispose()
312 {
313 if (_disposed)
314 {
315 return;
316 }
317
318 _exifTool?.Dispose();
319 _disposed = true;
320 }

764 public void GenerateTargetPaths(Action<string>? progressCallback = null)
765 {
766 ThrowIfDisposed();
767
768 _database.SortRecords(_configuration.FileSorting, _configuration.FileSortingOrder);
769
770 var pendingCount = _database.Count(record =>
771 !string.IsNullOrWhiteSpace(record.SourceFilePath) &&
772 record.Status is not (ProcessingStatus.PathGenerated or ProcessingStatus.Executed));
773
774 BeginStage(FIRECatalogStage.Generate, pendingCount);
775
776 foreach (var record in _database)
777 {
778 if (string.IsNullOrWhiteSpace(record.SourceFilePath)) continue;
779
780 // Skip files that already have a generated path or have been executed.
781 if (record.Status is ProcessingStatus.PathGenerated or ProcessingStatus.Executed) continue;
782
783 progressCallback?.Invoke(record.SourceFilePath);
784 ReportFileProgress(record.SourceFilePath);
785
786 // Handle sidecar files separately
787 if (record.Classification == FileClassification.SidecarFile)
788 {
789 GenerateSidecarTargetPath(record);
790 // Update status for sidecars
791 if (!string.IsNullOrWhiteSpace(record.TargetFilePath))
792 record.Status = ProcessingStatus.PathGenerated;
793 continue;
794 }
795
796 var extension = Path.GetExtension(record.SourceFilePath).ToLowerInvariant();
797 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig)) continue;
798
799 var metadataLookup = record.FileMetaDatas.ToDictionary(
800 m => m.Key ?? string.Empty,
801 m => m.Value ?? string.Empty,
802 StringComparer.OrdinalIgnoreCase);
803
804 var sortingPattern = extConfig.SortingPatern ?? _configuration.SortingPatern ?? string.Empty;
805 var fileNamePattern = extConfig.FileNamePatern ?? _configuration.FileNamePatern ?? string.Empty;
806
807 var targetDirectory = ParseTemplate(sortingPattern, metadataLookup, record.SourceFilePath);
808 if (string.IsNullOrWhiteSpace(targetDirectory)) continue;
809
810 long? counter = null;
811 if (ContainsCounterPlaceholder(fileNamePattern))
812 counter = _database.GetNextCounterValue(targetDirectory);
813
814 var targetFileName = ParseTemplate(fileNamePattern, metadataLookup, record.SourceFilePath, counter);
815
816 if (!string.IsNullOrWhiteSpace(targetFileName))
817 {
818 record.TargetFilePath = Path.Combine(targetDirectory, targetFileName);
819 // Update status after successful path generation
820 record.Status = ProcessingStatus.PathGenerated;
821 }
822 }
823
824 _database.Save();
825 CompleteCurrentStage();
826 }

847 private void GenerateSidecarTargetPath(FIREDbRecord sidecarRecord)
848 {
849 if (string.IsNullOrWhiteSpace(sidecarRecord.SourceFilePath))
850 return;
851
852 var sidecarDirectory = Path.GetDirectoryName(sidecarRecord.SourceFilePath);
853 var sidecarBaseName = Path.GetFileNameWithoutExtension(sidecarRecord.SourceFilePath);
854 var sidecarExtension = Path.GetExtension(sidecarRecord.SourceFilePath);
855
856 if (string.IsNullOrWhiteSpace(sidecarDirectory) || string.IsNullOrWhiteSpace(sidecarBaseName))
857 return;
858
859 // Find the primary file in the same directory with the same base name
860 var primaryRecord = _database.FirstOrDefault(r =>
861 r.Classification == FileClassification.RegularFile &&
862 !string.IsNullOrWhiteSpace(r.SourceFilePath) &&
863 string.Equals(Path.GetDirectoryName(r.SourceFilePath), sidecarDirectory, StringComparison.OrdinalIgnoreCase) &&
864 string.Equals(Path.GetFileNameWithoutExtension(r.SourceFilePath), sidecarBaseName, StringComparison.OrdinalIgnoreCase));
865
866 if (primaryRecord == null || string.IsNullOrWhiteSpace(primaryRecord.TargetFilePath))
867 {
868 LogMetadataWarning(sidecarRecord.SourceFilePath, "Sidecar", "could not find primary file or primary file has no target path");
869 return;
870 }
871
872 // Inherit target directory and base name from primary, keep sidecar extension
873 var primaryTargetDirectory = Path.GetDirectoryName(primaryRecord.TargetFilePath);
874 var primaryTargetBaseName = Path.GetFileNameWithoutExtension(primaryRecord.TargetFilePath);
875
876 if (!string.IsNullOrWhiteSpace(primaryTargetDirectory) && !string.IsNullOrWhiteSpace(primaryTargetBaseName))
877 {
878 sidecarRecord.TargetFilePath = Path.Combine(primaryTargetDirectory, primaryTargetBaseName + sidecarExtension);
879 }
880 }

907 public void ExecuteFileOperations(Action<string>? progressCallback = null)
908 {
909 ThrowIfDisposed();
910
911 var pendingCount = _database.Count(record =>
912 !string.IsNullOrWhiteSpace(record.SourceFilePath) &&
913 !string.IsNullOrWhiteSpace(record.TargetFilePath) &&
914 record.Status != ProcessingStatus.Executed &&
915 File.Exists(record.SourceFilePath));
916
917 BeginStage(FIRECatalogStage.Execute, pendingCount);
918
919 foreach (var record in _database)
920 {
921 if (string.IsNullOrWhiteSpace(record.SourceFilePath) || string.IsNullOrWhiteSpace(record.TargetFilePath)) continue;
922 if (!File.Exists(record.SourceFilePath)) continue;
923
924 // Skip files that have already been executed (incremental workflow)
925 if (record.Status == ProcessingStatus.Executed) continue;
926
927 progressCallback?.Invoke(record.SourceFilePath);
928 ReportFileProgress(record.SourceFilePath);
929
930 string action;
931
932 // Sidecar files inherit the action from their primary file
933 if (record.Classification == FileClassification.SidecarFile)
934 {
935 // Use global default action for sidecars
936 action = _configuration.Action ?? "Copy";
937 }
938 else
939 {
940 // Regular files use their extension-specific configuration
941 var extension = Path.GetExtension(record.SourceFilePath).ToLowerInvariant();
942 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig)) continue;
943 action = extConfig.Action ?? _configuration.Action ?? "Copy";
944 }
945
946 ExecuteAction(action, record.SourceFilePath, record.TargetFilePath);
947
948 // Mark as executed after successful operation
949 record.Status = ProcessingStatus.Executed;
950 }
951
952 _database.Save();
953 CompleteCurrentStage();
954 }

1040 private void ProcessFile(string filePath)
1041 {
1042 var extension = Path.GetExtension(filePath).ToLowerInvariant();
1043 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig))
1044 {
1045 if (CurrentStage == FIRECatalogStage.Collect)
1046 {
1047 var key = string.IsNullOrEmpty(extension) ? "(no extension)" : extension;
1048 _collectSkippedExtensions.TryGetValue(key, out var count);
1049 _collectSkippedExtensions[key] = count + 1;
1050 }
1051 return;
1052 }
1053
1054 var fileIdInfo = GetFileIdInfo(filePath);
1055
1056 // Skip files that are already in the database (incremental workflow)
1057 if (_database.FileExists(fileIdInfo.VolumeSerialNumber, fileIdInfo.FileId))
1058 return;
1059
1060 var record = new FIREDbRecord
1061 {
1062 SourceFilePath = filePath,
1063 VolumeSerialNumber = fileIdInfo.VolumeSerialNumber,
1064 FileId = fileIdInfo.FileId
1065 };
1066
1067 foreach (var keywordEntry in extConfig.AvailableKeyWords)
1068 {
1069 var keywordName = keywordEntry.Key;
1070 var keywordConfig = keywordEntry.Value;
1071 var sourceName = keywordConfig.Source ?? "FILEINFO";
1072 var valAttribute = keywordConfig.ValAttribute ?? "LOWEST";
1073
1074 var source = _metadataRegistry.GetSource(sourceName);
1075 if (source == null)
1076 {
1077 LogMetadataWarning(filePath, keywordName, $"metadata source '{sourceName}' was not found");
1078 var fallbackValue = ResolveMissingKeywordValue(keywordConfig, filePath, keywordName);
1079 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, fallbackValue, keywordConfig, sourceName));
1080 continue;
1081 }
1082
1083 if (keywordConfig.KeyWords.Count == 0)
1084 {
1085 LogMetadataWarning(filePath, keywordName, "no keywords configured");
1086 var fallbackValue = ResolveMissingKeywordValue(keywordConfig, filePath, keywordName);
1087 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, fallbackValue, keywordConfig, sourceName));
1088 continue;
1089 }
1090
1091 var extractedMetadata = source.ExtractMetadata(filePath);
1092 var matchingValues = new List<string>();
1093
1094 foreach (var keyword in keywordConfig.KeyWords)
1095 {
1096 if (extractedMetadata.TryGetValue(keyword, out var value) && !string.IsNullOrWhiteSpace(value))
1097 matchingValues.Add(value);
1098 }
1099
1100 if (matchingValues.Count == 0)
1101 {
1102 LogMetadataWarning(filePath, keywordName, $"none of the configured keywords were found in source '{sourceName}'");
1103 var fallbackValue = ResolveMissingKeywordValue(keywordConfig, filePath, keywordName);
1104 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, fallbackValue, keywordConfig, sourceName));
1105 continue;
1106 }
1107
1108 var selectedValue = SelectValue(matchingValues, valAttribute, keywordConfig.DataType, filePath, keywordName);
1109 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, selectedValue, keywordConfig, sourceName));
1110 }
1111
1112 _database.Add(record);
1113 if (CurrentStage == FIRECatalogStage.Collect)
1114 _collectAddedFiles++;
1115 ProcessSidecarFiles(filePath, extConfig, record);
1116 }

1141 private void ProcessSidecarFiles(string primaryFilePath, FileExtensionConfiguration extConfig, FIREDbRecord primaryRecord)
1142 {
1143 if (extConfig.SidecarFileExtensions == null || extConfig.SidecarFileExtensions.Count == 0)
1144 return;
1145
1146 var directory = Path.GetDirectoryName(primaryFilePath);
1147 if (string.IsNullOrWhiteSpace(directory) || !Directory.Exists(directory))
1148 return;
1149
1150 var fileNameWithoutExtension = Path.GetFileNameWithoutExtension(primaryFilePath);
1151
1152 foreach (var sidecarExtension in extConfig.SidecarFileExtensions)
1153 {
1154 var normalizedExtension = sidecarExtension.StartsWith(".", StringComparison.Ordinal)
1155 ? sidecarExtension
1156 : "." + sidecarExtension;
1157
1158 var sidecarPath = Path.Combine(directory, fileNameWithoutExtension + normalizedExtension);
1159
1160 if (!File.Exists(sidecarPath))
1161 continue;
1162
1163 try
1164 {
1165 var fileIdInfo = GetFileIdInfo(sidecarPath);
1166
1167 // Skip sidecar if already in database (incremental workflow)
1168 if (_database.FileExists(fileIdInfo.VolumeSerialNumber, fileIdInfo.FileId))
1169 continue;
1170
1171 var sidecarRecord = new FIREDbRecord
1172 {
1173 SourceFilePath = sidecarPath,
1174 VolumeSerialNumber = fileIdInfo.VolumeSerialNumber,
1175 FileId = fileIdInfo.FileId,
1176 Classification = FileClassification.SidecarFile,
1177 Status = primaryRecord.Status // Inherit status from primary file
1178 };
1179
1180 // Clone metadata from primary file so the sidecar can inherit the target path
1181 foreach (var metadata in primaryRecord.FileMetaDatas)
1182 {
1183 sidecarRecord.FileMetaDatas.Add(new FIREFileMetaData
1184 {
1185 Key = metadata.Key,
1186 Value = metadata.Value,
1187 TypeName = metadata.TypeName,
1188 DataSource = metadata.DataSource
1189 });
1190 }
1191
1192 _database.Add(sidecarRecord);
1193 }
1194 catch
1195 {
1196 // If sidecar processing fails, continue with next sidecar
1197 // The primary file is already in the database
1198 }
1199 }
1200 }

1228 private string ResolveMissingKeywordValue(AvailableKeywordConfiguration keywordConfig, string filePath, string keywordName)
1229 {
1230 var resolved = ResolveKeywordDefaultValue(keywordConfig, DateTime.Now);
1231 if (!string.Equals(resolved, "NA", StringComparison.Ordinal))
1232 {
1233 return resolved;
1234 }
1235
1236 if (!string.IsNullOrWhiteSpace(keywordConfig.Default) &&
1237 string.Equals((keywordConfig.DataType ?? "STRING").Trim(), "DATETIME", StringComparison.OrdinalIgnoreCase))
1238 {
1239 LogMetadataWarning(filePath, keywordName, $"configured DATETIME default '{keywordConfig.Default.Trim()}' is invalid; falling back to NA");
1240 }
1241
1242 return "NA";
1243 }

1245 internal static string ResolveKeywordDefaultValue(AvailableKeywordConfiguration keywordConfig, DateTime now)
1246 {
1247 if (string.IsNullOrWhiteSpace(keywordConfig.Default))
1248 {
1249 return "NA";
1250 }
1251
1252 var configuredDefault = keywordConfig.Default.Trim();
1253 var dataType = (keywordConfig.DataType ?? "STRING").Trim();
1254
1255 if (!dataType.Equals("DATETIME", StringComparison.OrdinalIgnoreCase))
1256 {
1257 return configuredDefault;
1258 }
1259
1260 if (configuredDefault.Equals("NOW", StringComparison.OrdinalIgnoreCase))
1261 {
1262 return now.ToString("yyyy:MM:dd HH:mm:ss", CultureInfo.InvariantCulture);
1263 }
1264
1265 if (TryNormalizeDateTime(configuredDefault, out var normalizedDate) &&
1266 DateTime.TryParseExact(
1267 normalizedDate,
1268 "yyyy:MM:dd HH:mm:ss",
1269 CultureInfo.InvariantCulture,
1270 DateTimeStyles.None,
1271 out _))
1272 {
1273 return normalizedDate;
1274 }
1275
1276 return "NA";
1277 }

1288 private string SelectValue(List<string> values, string valAttribute, string dataType, string filePath, string keywordName)
1289 {
1290 var selectionMode = valAttribute.Trim().ToUpperInvariant();
1291 var normalizedDataType = dataType.Trim().ToUpperInvariant();
1292

1503 internal static string ApplyStringReplacement(string input, string pattern, string replacement)
1504 {
1505 if (string.IsNullOrEmpty(input) || string.IsNullOrWhiteSpace(pattern))
1506 return input;
1507
1508 replacement ??= string.Empty;

1510 if (pattern.StartsWith("regex:", StringComparison.OrdinalIgnoreCase))
1511 {

1514 return input;
1515
1516 try
1517 {
1518 return Regex.Replace(input, regexPattern, replacement, RegexOptions.IgnoreCase | RegexOptions.CultureInvariant);
1519 }

1523 }
1524 }
