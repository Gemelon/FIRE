145public sealed class FileInfoMetadataSource : IMetadataSource

152
161 public Dictionary<string, string> ExtractMetadata(string filePath)
162 {
163 var metadata = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
164 try
165 {
166 var fileInfo = new FileInfo(filePath);
167 if (!fileInfo.Exists) return metadata;
168
169 metadata["CreationTime"] = fileInfo.CreationTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
170 metadata["ModificationTime"] = fileInfo.LastWriteTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
171 metadata["AccessTime"] = fileInfo.LastAccessTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
172 }
173 catch { }
174 return metadata;
175 }
176}

271 public Dictionary<string, string> ExtractMetadata(string filePath)
272 {
273 var metadata = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
274
275 try
276 {
277 if (!File.Exists(filePath))
278 {
279 return metadata;
280 }
281
282 // Execute ExifTool synchronously and get all metadata
283 var result = _exifTool.ExtractAllMetadataAsync(filePath).GetAwaiter().GetResult();
284
285 if (result != null)
286 {
287 foreach (var kvp in result)
288 {
289 if (!string.IsNullOrWhiteSpace(kvp.Key) && kvp.Value != null)
290 {
291 metadata[kvp.Key] = kvp.Value.ToString() ?? string.Empty;
292 }
293 }
294 }
295 }
296 catch
297 {
298 // Metadata extraction failed; return partial or empty result.
299 }
300
301 return metadata;
302 }

312 {
313 if (_disposed)
314 {
315 return;
316 }
317
318 _exifTool?.Dispose();

789 public void GenerateTargetPaths(Action<string>? progressCallback = null)
790 {
791 ThrowIfDisposed();
792
793 _database.SortRecords(_configuration.FileSorting, _configuration.FileSortingOrder);
794
795 var pendingCount = _database.Count(record =>
796 !string.IsNullOrWhiteSpace(record.SourceFilePath) &&
797 record.Status is not (ProcessingStatus.PathGenerated or ProcessingStatus.Executed) &&
798 !record.ExcludedFromGenerate);
799
800 BeginStage(FIRECatalogStage.Generate, pendingCount);
801
802 foreach (var record in _database)
803 {
804 if (string.IsNullOrWhiteSpace(record.SourceFilePath)) continue;
805
806 // Skip files that already have a generated path or have been executed.
807 if (record.Status is ProcessingStatus.PathGenerated or ProcessingStatus.Executed) continue;
808
809 // Skip files explicitly excluded from Generate phase
810 if (record.ExcludedFromGenerate)
811 {
812 _logger?.Log(FIRELogLevel.Info, "Generate", $"Skipped (excluded by user): {record.SourceFilePath}");
813 continue;
814 }
815
816 progressCallback?.Invoke(record.SourceFilePath);
817 ReportFileProgress(record.SourceFilePath);
818
819 // Handle sidecar files separately
820 if (record.Classification == FileClassification.SidecarFile)
821 {
822 GenerateSidecarTargetPath(record);
823 // Update status for sidecars
824 if (!string.IsNullOrWhiteSpace(record.TargetFilePath))
825 record.Status = ProcessingStatus.PathGenerated;
826 continue;
827 }
828
829 var extension = Path.GetExtension(record.SourceFilePath).ToLowerInvariant();
830 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig)) continue;
831
832 var metadataLookup = record.FileMetaDatas.ToDictionary(
833 m => m.Key ?? string.Empty,
834 m => m.Value ?? string.Empty,
835 StringComparer.OrdinalIgnoreCase);
836 var metadataTypeLookup = record.FileMetaDatas.ToDictionary(
837 m => m.Key ?? string.Empty,
838 m => m.TypeName ?? string.Empty,
839 StringComparer.OrdinalIgnoreCase);
840
841 var sortingPattern = extConfig.SortingPatern ?? _configuration.SortingPatern ?? string.Empty;
842 var fileNamePattern = extConfig.FileNamePatern ?? _configuration.FileNamePatern ?? string.Empty;
843
844 var targetDirectory = ParseTemplate(sortingPattern, metadataLookup, record.SourceFilePath, null, metadataTypeLookup);
845 if (string.IsNullOrWhiteSpace(targetDirectory)) continue;
846
847 long? counter = null;
848 if (ContainsCounterPlaceholder(fileNamePattern))
849 counter = _database.GetNextCounterValue(targetDirectory);
850
851 var targetFileName = ParseTemplate(fileNamePattern, metadataLookup, record.SourceFilePath, counter, metadataTypeLookup);
852
853 if (!string.IsNullOrWhiteSpace(targetFileName))
854 {
855 record.TargetFilePath = Path.Combine(targetDirectory, targetFileName);
856 // Update status after successful path generation
857 record.Status = ProcessingStatus.PathGenerated;
858 _logger?.LogContinuation(FIRELogLevel.Debug, record.TargetFilePath);
859 }
860 }
861
862 _database.Save();
863 CompleteCurrentStage();
864 }

885 private void GenerateSidecarTargetPath(FIREDbRecord sidecarRecord)
886 {
887 if (string.IsNullOrWhiteSpace(sidecarRecord.SourceFilePath))
888 return;
889
890 var sidecarDirectory = Path.GetDirectoryName(sidecarRecord.SourceFilePath);
891 var sidecarBaseName = Path.GetFileNameWithoutExtension(sidecarRecord.SourceFilePath);
892 var sidecarExtension = Path.GetExtension(sidecarRecord.SourceFilePath);
893
894 if (string.IsNullOrWhiteSpace(sidecarDirectory) || string.IsNullOrWhiteSpace(sidecarBaseName))
895 return;
896
897 // Find the primary file in the same directory with the same base name
898 var primaryRecord = _database.FirstOrDefault(r =>
899 r.Classification == FileClassification.RegularFile &&
900 !string.IsNullOrWhiteSpace(r.SourceFilePath) &&
901 string.Equals(Path.GetDirectoryName(r.SourceFilePath), sidecarDirectory, StringComparison.OrdinalIgnoreCase) &&
902 string.Equals(Path.GetFileNameWithoutExtension(r.SourceFilePath), sidecarBaseName, StringComparison.OrdinalIgnoreCase));
903
904 if (primaryRecord == null || string.IsNullOrWhiteSpace(primaryRecord.TargetFilePath))
905 {
906 LogMetadataWarning(sidecarRecord.SourceFilePath, "Sidecar", "could not find primary file or primary file has no target path");
907 return;
908 }
909
910 // Inherit target directory and base name from primary, keep sidecar extension
911 var primaryTargetDirectory = Path.GetDirectoryName(primaryRecord.TargetFilePath);
912 var primaryTargetBaseName = Path.GetFileNameWithoutExtension(primaryRecord.TargetFilePath);
913
914 if (!string.IsNullOrWhiteSpace(primaryTargetDirectory) && !string.IsNullOrWhiteSpace(primaryTargetBaseName))
915 {
916 sidecarRecord.TargetFilePath = Path.Combine(primaryTargetDirectory, primaryTargetBaseName + sidecarExtension);
917 }

945 public void ExecuteFileOperations(Action<string>? progressCallback = null)
946 {
947 ThrowIfDisposed();
948
949 var pendingCount = _database.Count(record =>
950 !string.IsNullOrWhiteSpace(record.SourceFilePath) &&
951 !string.IsNullOrWhiteSpace(record.TargetFilePath) &&
952 record.Status != ProcessingStatus.Executed &&
953 !record.ExcludedFromExecute &&
954 File.Exists(record.SourceFilePath));
955
956 BeginStage(FIRECatalogStage.Execute, pendingCount);
957
958 foreach (var record in _database)
959 {
960 if (string.IsNullOrWhiteSpace(record.SourceFilePath) || string.IsNullOrWhiteSpace(record.TargetFilePath)) continue;
961 if (!File.Exists(record.SourceFilePath)) continue;
962
963 // Skip files that have already been executed (incremental workflow)
964 if (record.Status == ProcessingStatus.Executed) continue;
965
966 // Skip files explicitly excluded from Execute phase
967 if (record.ExcludedFromExecute)
968 {
969 _logger?.Log(FIRELogLevel.Info, "Execute", $"Skipped (excluded by user): {record.SourceFilePath}");
970 continue;
971 }
972
973 // Skip files that don't have a generated target path (e.g., excluded from Generate)
974 if (record.Status == ProcessingStatus.NotProcessed)
975 {
976 _logger?.Log(FIRELogLevel.Info, "Execute", $"Skipped (not generated): {record.SourceFilePath}");
977 continue;
978 }
979
980 progressCallback?.Invoke(record.SourceFilePath);
981 ReportFileProgress(record.SourceFilePath);
982
983 string action;
984
985 // Sidecar files inherit the action from their primary file
986 if (record.Classification == FileClassification.SidecarFile)
987 {
988 // Use global default action for sidecars
989 action = _configuration.Action ?? "Copy";
990 }
991 else
992 {
993 // Regular files use their extension-specific configuration
994 var extension = Path.GetExtension(record.SourceFilePath).ToLowerInvariant();
995 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig)) continue;
996 action = extConfig.Action ?? _configuration.Action ?? "Copy";
997 }
998
999 ExecuteAction(action, record.SourceFilePath, record.TargetFilePath);
1000
1001 // Mark as executed after successful operation
1002 record.Status = ProcessingStatus.Executed;
1003 }
1004
1005 _database.Save();
1006 CompleteCurrentStage();
1007 }

1954 internal static string ApplyStringReplacement(string input, string pattern, string replacement)

1998 private string ResolvePlaceholder(
1999 string key,
2000 Dictionary<string, string> metadata,
2001 string sourceFilePath,
2002 long? counter = null,
2003 Dictionary<string, string>? metadataTypes = null)
2004 {
2005 var (baseName, suffix) = ParsePlaceholderExpression(key);
2006
2007 if (baseName.Equals("FileName", StringComparison.OrdinalIgnoreCase))
2008 {
2009 if (!string.IsNullOrWhiteSpace(suffix) && suffix.Equals("Noext", StringComparison.OrdinalIgnoreCase))
2010 return ApplyConfiguredStringReplacements(Path.GetFileNameWithoutExtension(sourceFilePath));
2011 return ApplyConfiguredStringReplacements(Path.GetFileName(sourceFilePath));
2012 }
2013
2014 if (baseName.Equals("MediaRootPath", StringComparison.OrdinalIgnoreCase))
2015 return ApplyConfiguredStringReplacements(_configuration.MediaRootPath);
2016
2017 if (baseName.Equals("RootPath", StringComparison.OrdinalIgnoreCase))
2018 return ParseTemplate(_configuration.RootPath, metadata, sourceFilePath, counter, metadataTypes);
2019
2020 if (baseName.Equals("Counter", StringComparison.OrdinalIgnoreCase))
2021 {
2022 var effectiveCounter = counter ?? 1L;
2023 if (!string.IsNullOrWhiteSpace(suffix) && suffix.Length > 1 && suffix.StartsWith("D", StringComparison.OrdinalIgnoreCase)
2024 && int.TryParse(suffix[1..], NumberStyles.None, CultureInfo.InvariantCulture, out var digits)
2025 && digits > 0)
2026 {
2027 return effectiveCounter.ToString($"D{digits}", CultureInfo.InvariantCulture);
2028 }
2029
2030 return effectiveCounter.ToString(CultureInfo.InvariantCulture);
2031 }
2032
2033 if (!metadata.TryGetValue(baseName, out var value) || string.IsNullOrWhiteSpace(value))
2034 return "Unknown";
2035
2036 if (string.IsNullOrWhiteSpace(suffix))
2037 return ApplyConfiguredStringReplacements(value);
2038
2039 string? dataType = null;
2040 if (metadataTypes != null)
2041 metadataTypes.TryGetValue(baseName, out dataType);
2042
2043 if (!IsDateTimeDataType(dataType))
2044 return ApplyConfiguredStringReplacements(value);
2045
2046 var resolvedDateTimeValue = ResolveDateTimePlaceholderValue(value, suffix, sourceFilePath, baseName, logFallback: true);
2047 return ApplyConfiguredStringReplacements(resolvedDateTimeValue);
2048 }
