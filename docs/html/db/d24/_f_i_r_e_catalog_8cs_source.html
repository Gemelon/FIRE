143public sealed class FileInfoMetadataSource : IMetadataSource
144{
149 public string SourceName => "FILEINFO";
150
159 public Dictionary<string, string> ExtractMetadata(string filePath)
160 {
161 var metadata = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
162 try
163 {
164 var fileInfo = new FileInfo(filePath);
165 if (!fileInfo.Exists) return metadata;
166
167 metadata["CreationTime"] = fileInfo.CreationTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
168 metadata["ModificationTime"] = fileInfo.LastWriteTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
169 metadata["AccessTime"] = fileInfo.LastAccessTime.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
170 }
171 catch { }
172 return metadata;
173 }
174}

269 public Dictionary<string, string> ExtractMetadata(string filePath)
270 {
271 var metadata = new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
272
273 try
274 {
275 if (!File.Exists(filePath))
276 {
277 return metadata;
278 }
279
280 // Execute ExifTool synchronously and get all metadata
281 var result = _exifTool.ExtractAllMetadataAsync(filePath).GetAwaiter().GetResult();
282
283 if (result != null)
284 {
285 foreach (var kvp in result)
286 {
287 if (!string.IsNullOrWhiteSpace(kvp.Key) && kvp.Value != null)
288 {
289 metadata[kvp.Key] = kvp.Value.ToString() ?? string.Empty;
290 }
291 }
292 }
293 }
294 catch
295 {
296 // Metadata extraction failed; return partial or empty result.
297 }
298
299 return metadata;
300 }

309 public void Dispose()
310 {
311 if (_disposed)
312 {
313 return;
314 }
315
316 _exifTool?.Dispose();
317 _disposed = true;

643 public void GenerateTargetPaths(Action<string>? progressCallback = null)
644 {
645 ThrowIfDisposed();
646 var directoryCounters = new Dictionary<string, int>(StringComparer.OrdinalIgnoreCase);
647
648 foreach (var record in _database)
649 {
650 if (string.IsNullOrWhiteSpace(record.SourceFilePath)) continue;
651
652 progressCallback?.Invoke(record.SourceFilePath);
653
654 var extension = Path.GetExtension(record.SourceFilePath);
655 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig)) continue;
656
657 var metadataLookup = record.FileMetaDatas.ToDictionary(
658 m => m.Key ?? string.Empty,
659 m => m.Value ?? string.Empty,
660 StringComparer.OrdinalIgnoreCase);
661
662 var sortingPattern = extConfig.SortingPatern ?? _configuration.SortingPatern ?? string.Empty;
663 var fileNamePattern = extConfig.FileNamePatern ?? _configuration.FileNamePatern ?? string.Empty;
664
665 var targetDirectory = ParseTemplate(sortingPattern, metadataLookup, record.SourceFilePath);
666 if (string.IsNullOrWhiteSpace(targetDirectory)) continue;
667
668 var counter = directoryCounters.TryGetValue(targetDirectory, out var currentCounter)
669 ? currentCounter + 1
670 : 1;
671 directoryCounters[targetDirectory] = counter;
672
673 var targetFileName = ParseTemplate(fileNamePattern, metadataLookup, record.SourceFilePath, counter);
674
675 if (!string.IsNullOrWhiteSpace(targetFileName))
676 record.TargetFilePath = Path.Combine(targetDirectory, targetFileName);
677 }
678 _database.Save();
679 }

762 private void ProcessFile(string filePath)
763 {
764 var extension = Path.GetExtension(filePath);
765 if (!_configuration.FileExtensions.TryGetValue(extension, out var extConfig)) return;
766
767 var fileIdInfo = GetFileIdInfo(filePath);
768 var record = new FIREDbRecord
769 {
770 SourceFilePath = filePath,
771 VolumeSerialNumber = fileIdInfo.VolumeSerialNumber,
772 FileId = fileIdInfo.FileId
773 };
774
775 foreach (var keywordEntry in extConfig.AvailableKeyWords)
776 {
777 var keywordName = keywordEntry.Key;
778 var keywordConfig = keywordEntry.Value;
779 var sourceName = keywordConfig.Source ?? "FILEINFO";
780 var valAttribute = keywordConfig.ValAttribute ?? "LOWEST";
781
782 var source = _metadataRegistry.GetSource(sourceName);
783 if (source == null)
784 {
785 LogMetadataWarning(filePath, keywordName, $"metadata source '{sourceName}' was not found");
786 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, "NA", keywordConfig, sourceName));
787 continue;
788 }
789
790 if (keywordConfig.KeyWords.Count == 0)
791 {
792 LogMetadataWarning(filePath, keywordName, "no keywords configured");
793 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, "NA", keywordConfig, sourceName));
794 continue;
795 }
796
797 var extractedMetadata = source.ExtractMetadata(filePath);
798 var matchingValues = new List<string>();
799
800 foreach (var keyword in keywordConfig.KeyWords)
801 {
802 if (extractedMetadata.TryGetValue(keyword, out var value) && !string.IsNullOrWhiteSpace(value))
803 matchingValues.Add(value);
804 }
805
806 if (matchingValues.Count == 0)
807 {
808 LogMetadataWarning(filePath, keywordName, $"none of the configured keywords were found in source '{sourceName}'");
809 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, "NA", keywordConfig, sourceName));
810 continue;
811 }
812
813 var selectedValue = SelectValue(matchingValues, valAttribute, keywordConfig.DataType, filePath, keywordName);
814 record.FileMetaDatas.Add(CreateMetadataEntry(keywordName, selectedValue, keywordConfig, sourceName));
815 }
816
817 _database.Add(record);
818 }

848 private static string SelectValue(List<string> values, string valAttribute, string dataType, string filePath, string keywordName)
849 {
850 var selectionMode = valAttribute.Trim().ToUpperInvariant();
851 var normalizedDataType = dataType.Trim().ToUpperInvariant();
852
853 // Early exit for single value
854 if (values.Count == 1)
855 {
856 // Normalize DateTime values if applicable
857 if (normalizedDataType is "DATETIME" or "DATE" or "TIME" &&
858 TryNormalizeDateTime(values[0], out var normalizedDate) &&
859 DateTime.TryParseExact(normalizedDate, "yyyy:MM:dd HH:mm:ss",
860 CultureInfo.InvariantCulture, DateTimeStyles.None, out var dateTimeValue))
861 {
862 return dateTimeValue.ToString();
863 }
864 return values[0];
865 }
866
867 // Strongly-typed data types: use SelectExtremeValue
868 if (normalizedDataType is "INT" or "INTEGER" or "DATETIME" or "DATE" or "TIME")
869 {
870 return selectionMode switch
871 {
872 "HIGHEST" => SelectExtremeValue(values, normalizedDataType, true, filePath, keywordName),
873 _ => SelectExtremeValue(values, normalizedDataType, false, filePath, keywordName)
874 };
875 }
876
877 // String comparison: use MaxBy/MinBy instead of OrderBy
878 return selectionMode switch
879 {
880 "HIGHEST" => values.MaxBy(value => value, StringComparer.OrdinalIgnoreCase) ?? "NA",
881 _ => values.MinBy(value => value, StringComparer.OrdinalIgnoreCase) ?? "NA"
882 };
883 }
884
894 private static string SelectExtremeValue(List<string> values, string normalizedDataType, bool highest, string filePath, string keywordName)
895 {
896 var candidates = new List<(string RawValue, IComparable ComparableValue)>(values.Count);
897
898 foreach (var value in values)
899 {
900 if (TryConvertComparable(value, normalizedDataType, out var comparableValue))
901 candidates.Add((value, comparableValue));
902 }
903
904 if (candidates.Count == 0)
905 {
906 LogMetadataWarning(filePath, keywordName, $"no comparable values found for data type '{normalizedDataType}'");
907 return values[0];
908 }
909
910 var selected = highest
911 ? candidates.MaxBy(x => x.ComparableValue)
912 : candidates.MinBy(x => x.ComparableValue);
913
914 return selected.ComparableValue.ToString();
915 }

940 private static bool TryNormalizeDateTime(string value, out string normalized)
941 {
942 if (DateTime.TryParseExact(
943 value.Trim(),
944 KnownDateTimeFormats,
945 CultureInfo.InvariantCulture,
946 DateTimeStyles.AllowWhiteSpaces | DateTimeStyles.AssumeLocal,
947 out var dt))
948 {
949 normalized = dt.ToString("yyyy:MM:dd HH:mm:ss", CultureInfo.InvariantCulture);
950 return true;
951 }
952
953 normalized = string.Empty;
954 return false;
955 }

964 private static bool TryConvertComparable(string value, string normalizedDataType, out IComparable comparableValue)
965 {
966 switch (normalizedDataType)
967 {
968 case "INT":
969 case "INTEGER":
970 if (long.TryParse(value, NumberStyles.Any, CultureInfo.InvariantCulture, out var longValue))
971 {
972 comparableValue = longValue;
973 return true;
974 }
975 break;
976 case "DATETIME":
977 case "DATE":
978 case "TIME":
979 if (TryNormalizeDateTime(value, out var normalizedDate) &&
980 DateTime.TryParseExact(normalizedDate, "yyyy:MM:dd HH:mm:ss",
981 CultureInfo.InvariantCulture, DateTimeStyles.None, out var dateTimeValue))
982 {
983 comparableValue = dateTimeValue;
984 return true;
985 }
986 break;
987 }
988
989 comparableValue = string.Empty;
990 return false;
991 }

1012 private string ParseTemplate(string template, Dictionary<string, string> metadata, string sourceFilePath, int? counter = null)
1013 {
1014 if (string.IsNullOrWhiteSpace(template)) return string.Empty;
1015
1016 var result = template;
1017 var placeholderPattern = new Regex(@"\{(?<key>[^}]+)\}", RegexOptions.Compiled);
1018
1019 result = placeholderPattern.Replace(result, match =>
1020 ResolvePlaceholder(match.Groups["key"].Value, metadata, sourceFilePath, counter));
1021
1022 foreach (var replacement in _configuration.StringReplacements)
1023 result = result.Replace(replacement.Key, replacement.Value, StringComparison.OrdinalIgnoreCase);
1024
1025 return result;
1026 }

1036 private string ResolvePlaceholder(string key, Dictionary<string, string> metadata, string sourceFilePath, int? counter = null)
1037 {
1038 var parts = key.Split('.');

1040 var property = parts.Length > 1 ? parts[1] : null;
1041
1042 if (baseName.StartsWith("Counter:", StringComparison.OrdinalIgnoreCase))
1043 {
1044 property = baseName["Counter:".Length..];

1046 }
1047
1048 if (baseName.Equals("FileName", StringComparison.OrdinalIgnoreCase))
1049 {
1050 if (property != null && property.Equals("Noext", StringComparison.OrdinalIgnoreCase))
1051 return Path.GetFileNameWithoutExtension(sourceFilePath);
1052 return Path.GetFileName(sourceFilePath);
1053 }
1054
1055 if (baseName.Equals("MediaRootPath", StringComparison.OrdinalIgnoreCase))
1056 return _configuration.MediaRootPath;
1057
1058 if (baseName.Equals("RootPath", StringComparison.OrdinalIgnoreCase))
1059 return ParseTemplate(_configuration.RootPath, metadata, sourceFilePath, counter);
1060
1061 if (baseName.Equals("Counter", StringComparison.OrdinalIgnoreCase))
1062 {
1063 var effectiveCounter = counter ?? 1;
1064 if (!string.IsNullOrWhiteSpace(property) && property.Length > 1 && property.StartsWith("D", StringComparison.OrdinalIgnoreCase)
1065 && int.TryParse(property[1..], NumberStyles.None, CultureInfo.InvariantCulture, out var digits)
1066 && digits > 0)
1067 {
1068 return effectiveCounter.ToString($"D{digits}", CultureInfo.InvariantCulture);
