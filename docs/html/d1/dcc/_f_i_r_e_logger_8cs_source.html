102 internal FIRELogger(LoggingConfiguration config)
103 {
104 _config = config;
105 _minLevel = ParseLevel(config.LogLevel);
106
107 Directory.CreateDirectory(config.LogFilePath);
108 PurgeOldFiles();
109 }

117 internal void Log(FIRELogLevel level, string? stage, string message)
118 {
119 if (level < _minLevel) return;
120
121 var stageTag = stage ?? "misc";
122 var now = DateTime.Now;
123 var today = DateOnly.FromDateTime(now);
124 var timestamp = now.ToString("yyyy-MM-dd HH:mm:ss", CultureInfo.InvariantCulture);
125 var safeMessage = message.Replace("|", "\\|");
126
127 lock (_lock)
128 {
129 EnsureWriter(stageTag, today);
130 _writer!.WriteLine($"| {timestamp} | {level,-8} | {stageTag,-4} | {safeMessage} |");
131 _writer.Flush();
132
133 RotateIfNeeded(stageTag, today);
134 }
135 }
136
150 internal void LogContinuation(FIRELogLevel level, string message)
151 {
152 if (level < _minLevel) return;
153
154 var safeMessage = message.Replace("|", "\\|");
155
156 lock (_lock)
157 {
158 if (_writer == null) return;
159 _writer.WriteLine($"| | | | \u2192 {safeMessage} |");
160 _writer.Flush();
161 }
162 }
163
164 // Opens a writer for the correct file, switching when the stage or date changes.
165 private void EnsureWriter(string stageTag, DateOnly today)
166 {
167 if (_writer != null && stageTag == _currentStageTag && today == _currentDate)
168 return;
169
170 CloseWriter();
171 _currentStageTag = stageTag;
172 _currentDate = today;
173 _segmentIndex = 0;
174 OpenSegment();
175 }
176
177 // Creates/appends the current segment file and writes the header if the file is new.
178 private void OpenSegment()
179 {
180 _currentFilePath = BuildPath(_currentStageTag, _currentDate, _segmentIndex);
181 var isNew = !File.Exists(_currentFilePath);
182
183 var stream = new FileStream(_currentFilePath, FileMode.Append, FileAccess.Write, FileShare.Read);
184 _writer = new StreamWriter(stream, System.Text.Encoding.UTF8, bufferSize: 4096, leaveOpen: false);
185
186 if (isNew)
187 {
188 _writer.WriteLine($"# FIRE Log — {_currentStageTag.ToUpperInvariant()} — {_currentDate:yyyy-MM-dd}");
189 _writer.WriteLine();
190 _writer.WriteLine(TableHeader);
191 }
192 }
193
194 // Rotates to the next segment when the current file exceeds the size limit.
195 private void RotateIfNeeded(string stageTag, DateOnly today)
196 {
197 if (!File.Exists(_currentFilePath)) return;
198 if (new FileInfo(_currentFilePath).Length < _config.MaxFileSizeBytes) return;
199
200 CloseWriter();
201 _segmentIndex++;
202 _currentStageTag = stageTag;
203 _currentDate = today;
204 OpenSegment();
205 }
206
207 // Builds the file path for the given stage, date, and segment index.
208 private string BuildPath(string stageTag, DateOnly date, int segment)
209 {
210 var datePart = date.ToString("yyyyMMdd", CultureInfo.InvariantCulture);
211 var segmentPart = segment == 0 ? string.Empty : $".{segment}";
212 var name = $"{_config.LogFileName}.{stageTag}.{datePart}{segmentPart}.md";
213 return Path.Combine(_config.LogFilePath, name);
214 }
215
216 // Deletes files in the log directory whose write time is older than MaxAgeDays.
217 private void PurgeOldFiles()
218 {
219 if (_config.MaxAgeDays <= 0) return;
220
221 var cutoff = DateTime.Now.AddDays(-_config.MaxAgeDays);
222 try
223 {
224 foreach (var file in Directory.EnumerateFiles(_config.LogFilePath, "*.md"))
225 {
226 if (File.GetLastWriteTime(file) < cutoff)
227 File.Delete(file);
228 }
229 }
230 catch { /* Best-effort purge; do not disrupt the run. */ }
231 }
232
233 private void CloseWriter()
234 {
235 _writer?.Flush();
236 _writer?.Dispose();
237 _writer = null;
238 }
239
243 public void Dispose()
244 {
245 if (_disposed) return;
246 lock (_lock)
247 {
248 CloseWriter();
249 }
250 _disposed = true;
251 }

253 // Parses a log-level string to the corresponding enum value, defaulting to Error.
254 internal static FIRELogLevel ParseLevel(string levelName) => levelName.Trim().ToUpperInvariant() switch
255 {
256 "DEBUG" => FIRELogLevel.Debug,
257 "INFO" => FIRELogLevel.Info,
258 "WARNING" => FIRELogLevel.Warning,
259 "ERROR" => FIRELogLevel.Error,
260 "CRITICAL" => FIRELogLevel.Critical,
261 _ => FIRELogLevel.Error,
262 };
263

266 {
267 FIRECatalogMessageLevel.Trace => FIRELogLevel.Debug,
268 FIRECatalogMessageLevel.Info => FIRELogLevel.Info,
269 FIRECatalogMessageLevel.Warning => FIRELogLevel.Warning,
270 FIRECatalogMessageLevel.Error => FIRELogLevel.Error,
271 _ => FIRELogLevel.Info,
272 };
