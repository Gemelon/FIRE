300 public static FIREConfigration Parse(string yaml)
301 {
302 ArgumentException.ThrowIfNullOrWhiteSpace(yaml);
303
304 var deserializer = new DeserializerBuilder()
305 .WithNamingConvention(NullNamingConvention.Instance)
306 .IgnoreUnmatchedProperties()
307 .Build();
308
309 var configuration = deserializer.Deserialize<FIREConfigration>(yaml) ?? new FIREConfigration();
310 configuration.Normalize();
311 return configuration;
312 }

343 private void Normalize()
344 {
345 DataBasePath ??= string.Empty;
346 DataBaseFileName ??= string.Empty;
347 Action = string.IsNullOrWhiteSpace(Action) ? "Copy" : Action;
348 MediaRootPath = string.IsNullOrWhiteSpace(MediaRootPath) ? "CC" : MediaRootPath;
349 RootPath = string.IsNullOrWhiteSpace(RootPath) ? "{MediaRootPath}" : RootPath;
350 SortingPatern ??= string.Empty;
351 FileNamePatern ??= string.Empty;
352 FileSorting ??= [];
353 FileSortingOrder = string.IsNullOrWhiteSpace(FileSortingOrder) ? "Ascending" : FileSortingOrder;
354 StringReplacements ??= new Dictionary<string, string>(StringComparer.OrdinalIgnoreCase);
355 FileExtensions ??= new Dictionary<string, FileExtensionConfiguration>(StringComparer.OrdinalIgnoreCase);
356
357 foreach (var item in FileExtensions.Values)
358 {
359 item.Normalize();
360 }
361 }
