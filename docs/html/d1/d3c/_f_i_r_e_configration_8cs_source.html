293 public static FIREConfigration Parse(string yaml)
294 {
295 ArgumentException.ThrowIfNullOrWhiteSpace(yaml);
296
297 var deserializer = new DeserializerBuilder()
298 .WithNamingConvention(NullNamingConvention.Instance)
299 .IgnoreUnmatchedProperties()
300 .Build();
301
302 var configuration = deserializer.Deserialize<FIREConfigration>(yaml) ?? new FIREConfigration();
303 configuration.Normalize();
304 return configuration;
305 }

337 {

341 MediaRootPath = string.IsNullOrWhiteSpace(MediaRootPath) ? "CC" : MediaRootPath;
342 RootPath = string.IsNullOrWhiteSpace(RootPath) ? "{MediaRootPath}" : RootPath;

348 FileExtensions ??= new Dictionary<string, FileExtensionConfiguration>(StringComparer.OrdinalIgnoreCase);
