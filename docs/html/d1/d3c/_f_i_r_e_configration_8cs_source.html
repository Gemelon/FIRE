310 public static FIREConfigration Parse(string yaml)
311 {
312 ArgumentException.ThrowIfNullOrWhiteSpace(yaml);
313
314 var deserializer = new DeserializerBuilder()
315 .WithNamingConvention(NullNamingConvention.Instance)
316 .IgnoreUnmatchedProperties()
317 .Build();
318
319 var configuration = deserializer.Deserialize<FIREConfigration>(yaml) ?? new FIREConfigration();
320 configuration.Normalize();
321 return configuration;

704/// Configures the FIRE logging subsystem.
705/// </summary>
706/// <remarks>
707/// <para>
708/// <strong>Activation:</strong>
709/// Logging is active only when this section is present in the YAML configuration.
710/// If the <c>Logging</c> key is absent, no log files are created.
711/// </para>
712/// <para>
713/// <strong>File naming:</strong>
714/// One file per calendar day and per stage is created:
715/// <c>{LogFileName}.col.{yyyyMMdd}.md</c> (collect),
716/// <c>{LogFileName}.gen.{yyyyMMdd}.md</c> (generate),
717/// <c>{LogFileName}.exe.{yyyyMMdd}.md</c> (execute).
718/// </para>
719/// <para>
720/// <strong>Rotation:</strong>
721/// When a log file exceeds <see cref="MaxFileSizeBytes"/> (default 10 MB), a new numbered
722/// segment is started automatically: <c>{LogFileName}.col.{yyyyMMdd}.1.md</c>, <c>.2.md</c>, …
723/// A single 10 MB limit keeps individual files easily openable in any text editor.
724/// </para>
725/// <para>
726/// <strong>Retention:</strong>
727/// On every catalog run, log files older than <see cref="MaxAgeDays"/> days (default 30) are
728/// deleted from the log directory. 30 days covers a reasonable audit window without
729/// consuming excessive disk space; adjust downwards for high-volume environments.
730/// </para>
731/// </remarks>
732public sealed class LoggingConfiguration
733{
740 [YamlMember(Alias = "LogFilePath")]
741 public string LogFilePath { get; set; } = string.Empty;
742
749 [YamlMember(Alias = "LogFileName")]
750 public string LogFileName { get; set; } = "FireLog";
751
759 [YamlMember(Alias = "LogLevel")]
760 public string LogLevel { get; set; } = "Error";
761
770 [YamlMember(Alias = "MaxFileSizeBytes")]
771 public long MaxFileSizeBytes { get; set; } = 10 * 1024 * 1024;
772
781 [YamlMember(Alias = "MaxAgeDays")]
782 public int MaxAgeDays { get; set; } = 30;
783
788 internal void Normalize(string databasePath)
789 {
790 LogFileName = string.IsNullOrWhiteSpace(LogFileName) ? "FireLog" : LogFileName;
791 LogLevel = string.IsNullOrWhiteSpace(LogLevel) ? "Error" : LogLevel;
792 MaxFileSizeBytes = MaxFileSizeBytes <= 0 ? 10 * 1024 * 1024 : MaxFileSizeBytes;
793 MaxAgeDays = MaxAgeDays < 0 ? 30 : MaxAgeDays;
794
795 if (string.IsNullOrWhiteSpace(LogFilePath))
796 {
797 LogFilePath = string.IsNullOrWhiteSpace(databasePath)
798 ? Path.Combine(AppContext.BaseDirectory, "Logs")
799 : Path.Combine(databasePath, "Logs");
800 }
801 }
