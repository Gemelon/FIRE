8// Simple command-line argument parsing
9if (args.Length == 0)
10{
11 ShowHelp();
12 return 1;
13}
14
15var command = args[0].ToLowerInvariant();
16string? configPath = null;
17string? culture = null;
18bool clearDatabase = false;
19
20Console.CancelKeyPress += (_, e) =>
21{
22 e.Cancel = true;
23 AppLifetime.IsCancellationRequested = true;
24 Console.ForegroundColor = ConsoleColor.Yellow;
25 Console.WriteLine("\nCTRL+C detected. Stopping gracefully...");
26 Console.ResetColor();
27};
28
29// Parse arguments
30for (int i = 1; i < args.Length; i++)
31{
32 if ((args[i] == "--config" || args[i] == "-c") && i + 1 < args.Length)
33 {
34 configPath = args[++i];
35 }
36 else if ((args[i] == "--culture" || args[i] == "-l") && i + 1 < args.Length)
37 {
38 culture = args[++i];
39 }
40 else if (args[i] == "--clear-database" || args[i] == "--clear")
41 {
42 clearDatabase = true;
43 }
44}
45
46// Validate required arguments
47if (string.IsNullOrWhiteSpace(configPath))
48{
49 Console.WriteLine("ERROR: --config parameter is required");
50 ShowHelp();
51 return 1;
52}
53
54if (string.IsNullOrWhiteSpace(culture))
55{
56 Console.WriteLine("ERROR: --culture parameter is required");
57 ShowHelp();
58 return 1;
59}
60
61// Execute command
62return command switch
63{
64 "collect" => ExecuteCollect(configPath, culture, clearDatabase),
65 "generate" => ExecuteGenerate(configPath, culture),
66 "execute" => ExecuteOperations(configPath, culture),
67 "inspect" => ExecuteInspect(args, configPath, culture),
68 _ => ShowHelp()
69};

105static int ExecuteCollect(string configPath, string culture, bool clearDatabase)
106{
107 try
108 {
109 Console.WriteLine($"=== FIRE Collect Files ===");
110 Console.WriteLine($"Config: {configPath}");
111 Console.WriteLine($"Culture: {culture}");
112 if (clearDatabase)
113 Console.WriteLine("Database will be cleared before collecting.");
114 Console.WriteLine();
115
116 // Set culture
117 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = new CultureInfo(culture);
118
119 // Load configuration
120 Console.WriteLine("Loading configuration...");
121 var config = FIREConfigration.Load(configPath);
122 config.EnsureSupportedConfigurationVersion();
123 Console.WriteLine($"Configuration loaded successfully (Version: {config.ConfigurationVersion})");
124 Console.WriteLine();
125
126 // Create database path
127 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
128 Console.WriteLine($"Database: {dbPath}");
129
130 // Create catalog
131 Console.WriteLine("Initializing catalog...");
132 using var database = new FIREDatabase(dbPath);
133 using var catalog = new FIRECatalog(config, database);
134 Console.WriteLine("Catalog initialized.");
135 Console.WriteLine();
136
137 // Clear database if requested
138 if (clearDatabase)
139 {
140 Console.WriteLine("Clearing database...");
141 catalog.ClearDatabase();
142 Console.WriteLine("Database cleared.");
143 Console.WriteLine();
144 }
145
146 // Collect files
147 Console.WriteLine("Collecting files...");
148 var stopwatch = System.Diagnostics.Stopwatch.StartNew();
149 int fileCount = 0;
150 catalog.CollectFiles(filePath =>
151 {
152 AppLifetime.ThrowIfCancellationRequested();
153 fileCount++;
154 Console.Write($"\r[{fileCount}] Processing: {TruncatePath(filePath, 80)}".PadRight(Console.WindowWidth - 1));
155 });
156 stopwatch.Stop();
157 Console.WriteLine();
158 Console.WriteLine($"File collection completed in {stopwatch.Elapsed.TotalSeconds:F2} seconds.");
159 Console.WriteLine($"Total records: {database.Count}");
160 Console.WriteLine("Warnings: missing metadata entries are reported inline with [WARN].");
161 Console.WriteLine();
162
163 Console.WriteLine("Use the database to inspect collected files.");
164 return 0;
165 }
166 catch (OperationCanceledException)
167 {
168 Console.ForegroundColor = ConsoleColor.Yellow;
169 Console.WriteLine("Operation canceled by user (CTRL+C).");
170 Console.ResetColor();
171 return 2;
172 }
173 catch (Exception ex)
174 {
175 Console.ForegroundColor = ConsoleColor.Red;
176 Console.WriteLine($"ERROR: {ex.Message}");
177 Console.ResetColor();
178 Console.WriteLine(ex.StackTrace);
179 return 1;
180 }
181}

317static int ExecuteInspect(string[] args, string configPath, string culture)
318{
319 try
320 {
321 // Parse additional arguments for inspect command
322 string? filePath = null;
323 string? outputPath = null;
324 bool copyPath = false;
325
326 for (int i = 1; i < args.Length; i++)
327 {
328 if ((args[i] == "--file" || args[i] == "-f") && i + 1 < args.Length)
329 {
330 filePath = args[++i];
331 }
332 else if ((args[i] == "--output" || args[i] == "-o") && i + 1 < args.Length)
333 {
334 outputPath = args[++i];
335 }
336 else if (args[i] == "--copy-path")
337 {
338 copyPath = true;
339 }
340 }
341
342 if (string.IsNullOrWhiteSpace(filePath))
343 {
344 Console.ForegroundColor = ConsoleColor.Red;
345 Console.WriteLine("ERROR: --file parameter is required for inspect command");

351
352 if (!File.Exists(filePath))
353 {
354 Console.ForegroundColor = ConsoleColor.Red;
355 Console.WriteLine($"ERROR: File not found: {filePath}");
356 Console.ResetColor();
357 return 1;
358 }
359
360 Console.WriteLine($"=== FIRE Inspect File Metadata ===");
361 Console.WriteLine($"Config: {configPath}");
362 Console.WriteLine($"Culture: {culture}");
363 Console.WriteLine($"File: {filePath}");
364 if (!string.IsNullOrWhiteSpace(outputPath))
365 {
366 Console.WriteLine($"Output: {outputPath}");
367 }
368 Console.WriteLine();
369
370 // Set culture
371 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = new CultureInfo(culture);
372
373 // Load configuration
374 Console.WriteLine("Loading configuration...");
375 var config = FIREConfigration.Load(configPath);
376 config.EnsureSupportedConfigurationVersion();
377 Console.WriteLine($"Configuration loaded successfully (Version: {config.ConfigurationVersion})");
378 Console.WriteLine();
379
380 // Create database path (temporary, needed for catalog initialization)
381 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
382
383 // Create catalog
384 Console.WriteLine("Initializing catalog...");
385 using var database = new FIREDatabase(dbPath);
386 using var catalog = new FIRECatalog(config, database);
387 Console.WriteLine("Catalog initialized.");
388 Console.WriteLine();
389
390 // Extract and display metadata
391 Console.WriteLine("Extracting metadata...");
392 var stopwatch = System.Diagnostics.Stopwatch.StartNew();
393 var metadata = catalog.GetAllAvailableMetadata(filePath);
394 stopwatch.Stop();
395 Console.WriteLine($"Metadata extracted in {stopwatch.Elapsed.TotalMilliseconds:F2} ms.");
396 Console.WriteLine($"Total entries found: {metadata.Count}");
397 Console.WriteLine();
398
399 // Display metadata summary
400 if (metadata.Count > 0)
401 {
402 var groupedBySource = metadata.GroupBy(m => m.Source).OrderBy(g => g.Key);
403 Console.WriteLine("Metadata by Source:");
404 foreach (var sourceGroup in groupedBySource)
405 {
406 Console.WriteLine($" - {sourceGroup.Key}: {sourceGroup.Count()} entries");
407 }
408 Console.WriteLine();
409 }
410 else
411 {
412 Console.ForegroundColor = ConsoleColor.Yellow;
413 Console.WriteLine("WARNING: No metadata could be extracted from this file.");
414 Console.ResetColor();
415 Console.WriteLine();
416 }
417
418 // Write to Markdown
419 Console.WriteLine("Writing Markdown report...");
420 catalog.WriteMetadataToMarkdown(filePath, outputPath);
421
422 var finalOutputPath = outputPath ?? Path.Combine(
423 Path.GetDirectoryName(filePath) ?? ".",
424 Path.GetFileNameWithoutExtension(filePath) + ".md");
425 finalOutputPath = Path.GetFullPath(finalOutputPath);
426
427 Console.ForegroundColor = ConsoleColor.Green;
428 Console.WriteLine($"✓ Markdown report created: {finalOutputPath}");
429 Console.ResetColor();
430
431 if (copyPath)
432 {
433 if (TryCopyToClipboard(finalOutputPath))
434 {
435 Console.ForegroundColor = ConsoleColor.Cyan;
436 Console.WriteLine("✓ Markdown path copied to clipboard.");
437 Console.ResetColor();
438 }
439 else
440 {
441 Console.ForegroundColor = ConsoleColor.Yellow;
442 Console.WriteLine("WARNING: Could not copy Markdown path to clipboard.");
443 Console.ResetColor();
444 }
445 }
446
447 Console.WriteLine();
448
449 return 0;
450 }
451 catch (OperationCanceledException)
452 {
453 Console.ForegroundColor = ConsoleColor.Yellow;

455 Console.ResetColor();

468static bool TryCopyToClipboard(string text)
469{
470 if (string.IsNullOrWhiteSpace(text) || !OperatingSystem.IsWindows())
471 {
472 return false;
473 }
474
475 try
476 {
477 var startInfo = new ProcessStartInfo
478 {
479 FileName = "clip",
480 RedirectStandardInput = true,
481 UseShellExecute = false,
482 CreateNoWindow = true
483 };
484
485 using var process = Process.Start(startInfo);
486 if (process is null)
487 {
488 return false;
489 }
490
491 process.StandardInput.Write(text);
492 process.StandardInput.Close();
493 process.WaitForExit(2000);
494
495 return process.ExitCode == 0;
496 }
497 catch
498 {
499 return false;
500 }
501}

506static string TruncatePath(string path, int maxLength)
507{
508 if (string.IsNullOrEmpty(path) || path.Length <= maxLength)
509 return path;
510
511 // Try to show the end of the path which is usually more relevant
512 return "..." + path.Substring(path.Length - maxLength + 3);
513}
