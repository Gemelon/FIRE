12internal static class ProgramHost
13{
14 public static int Run(string[] args)
15 {
16 var bootstrapCulture = CultureResolver.ResolveCultureCode(CultureResolver.ReadCultureArgument(args) ?? "en-US");
17 var bootstrapLanguage = CultureResolver.ResolveLanguage(bootstrapCulture);
18
19 AppLifetime.CurrentLanguage = bootstrapLanguage;
20
21 Console.CancelKeyPress += (_, eventArgs) =>
22 {
23 eventArgs.Cancel = true;
24 AppLifetime.IsCancellationRequested = true;
25 ConsoleUi.WriteWarning(AppLifetime.CurrentLanguage, AppLifetime.NoWrap, TextCatalog.Get(AppLifetime.CurrentLanguage, "cancel_requested"));
26 };
27
28 var app = new CommandApp();
29 app.Configure(configuration =>
30 {
31 configuration.SetApplicationName("FIRE.Console");
32 configuration.ValidateExamples();
33
34 configuration
35 .AddCommand<CollectCommand>("collect")
36 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_collect"))
37 .WithExample("collect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "en-US")
38 .WithExample("collect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "de-DE", "--clear-database")
39 .WithExample("collect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "fr-FR", "--no-wrap");
40
41 configuration
42 .AddCommand<GenerateCommand>("generate")
43 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_generate"))
44 .WithExample("generate", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "en-US");
45
46 configuration
47 .AddCommand<ExecuteCommand>("execute")
48 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_execute"))
49 .WithExample("execute", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "de-DE");
50
51 configuration
52 .AddCommand<InspectCommand>("inspect")
53 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_inspect"))
54 .WithExample("inspect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "en-US", "--file", "image.jpg")
55 .WithExample("inspect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "fil-PH", "--file", "image.jpg", "--copy-path");
56 });
57
58 try
59 {
60 return app.Run(args);
61 }
62 catch (Exception ex)
63 {
64 ConsoleUi.WriteError(bootstrapLanguage, false, $"{TextCatalog.Get(bootstrapLanguage, "error_prefix")} {ex.Message}");
65 Console.WriteLine(ex.StackTrace);
66 return 1;
67 }
68 }

200 public static int ExecuteInspect(InspectSettings settings, RuntimeContext runtime)
201 {
202 try
203 {
204 if (string.IsNullOrWhiteSpace(settings.ConfigPath))
205 {
206 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_config_required"));
207 return 1;
208 }
209
210 if (string.IsNullOrWhiteSpace(settings.FilePath))
211 {
212 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_file_required"));
213 return 1;
214 }
215
216 var filePath = settings.FilePath;
217 if (!File.Exists(filePath))
218 {
219 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_file_not_found")} {filePath}");
220 return 1;
221 }
222
223 var normalizedCulture = runtime.CultureCode;
224 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = CultureInfo.GetCultureInfo(normalizedCulture);
225
226 ConsoleUi.WriteTitle(runtime, TextCatalog.Get(runtime.Language, "title_inspect"));
227 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_config")} {settings.ConfigPath}");
228 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_culture")} {normalizedCulture}");
229 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_file")} {filePath}");
230 if (!string.IsNullOrWhiteSpace(settings.OutputPath))
231 {
232 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_output")} {settings.OutputPath}");
233 }
234
235 ConsoleUi.WriteEmptyLine();
236 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "loading_configuration"));
237 var config = FIREConfigration.Load(settings.ConfigPath!);
238 config.EnsureSupportedConfigurationVersion();
239 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "configuration_loaded"), config.ConfigurationVersion));
240 ConsoleUi.WriteEmptyLine();
241
242 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
243 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_database")} {dbPath}");
244 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "initializing_catalog"));
245
246 using var database = new FIREDatabase(dbPath);
247 using var catalog = new FIRECatalog(config, database)
248 {
249 Culture = CultureInfo.CurrentUICulture
250 };
251
252 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "catalog_initialized"));
253 ConsoleUi.WriteEmptyLine();
254
255 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "extracting_metadata"));
256 var stopwatch = Stopwatch.StartNew();
257 var metadata = catalog.GetAllAvailableMetadata(filePath);
258 stopwatch.Stop();
259
260 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "metadata_extracted_ms"), stopwatch.Elapsed.TotalMilliseconds));
261 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "metadata_entries_found"), metadata.Count));
262 ConsoleUi.WriteEmptyLine();
263
264 if (metadata.Count > 0)
265 {
266 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "metadata_by_source"));
267 foreach (var sourceGroup in metadata.GroupBy(entry => entry.Source).OrderBy(group => group.Key))
268 {
269 ConsoleUi.WriteLine(runtime, $" - {sourceGroup.Key}: {sourceGroup.Count()}");
270 }
271 ConsoleUi.WriteEmptyLine();
272 }
273 else
274 {
275 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "warn_no_metadata"));
276 ConsoleUi.WriteEmptyLine();
277 }
278
279 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "writing_markdown"));
280 catalog.WriteMetadataToMarkdown(filePath, settings.OutputPath);
281
282 var finalOutputPath = settings.OutputPath ?? Path.Combine(
283 Path.GetDirectoryName(filePath) ?? ".",
284 Path.GetFileNameWithoutExtension(filePath) + ".md");
285 finalOutputPath = Path.GetFullPath(finalOutputPath);
286
287 ConsoleUi.WriteSuccess(runtime, $"{TextCatalog.Get(runtime.Language, "markdown_created")} {finalOutputPath}");
288
289 if (settings.CopyPath)
290 {
291 if (TryCopyToClipboard(finalOutputPath))
292 {
293 ConsoleUi.WriteInfo(runtime, TextCatalog.Get(runtime.Language, "markdown_copied"));
294 }
295 else
296 {
297 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "warn_copy_failed"));
298 }
299 }
300
301 ConsoleUi.WriteEmptyLine();
302 return 0;
303 }
304 catch (OperationCanceledException)
305 {
306 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "operation_canceled"));
307 return 2;
308 }
309 catch (Exception ex)
310 {
311 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
312 Console.WriteLine(ex.StackTrace);
313 return 1;
314 }
315 }

317 private static int ExecuteCatalogPipeline(
318 CommonCommandSettings settings,
319 RuntimeContext runtime,
320 string titleKey,
321 string progressLineKey,
322 string progressVerbKey,
323 string summaryKey,
324 string summaryHintKey,
325 Action<FIRECatalog> beforeAction,
326 Action<FIRECatalog, Action<string>> action)
327 {
328 if (string.IsNullOrWhiteSpace(settings.ConfigPath))
329 {
330 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_config_required"));
331 return 1;
332 }
333
334 var normalizedCulture = runtime.CultureCode;
335 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = CultureInfo.GetCultureInfo(normalizedCulture);
336
337 ConsoleUi.WriteTitle(runtime, TextCatalog.Get(runtime.Language, titleKey));
338 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_config")} {settings.ConfigPath}");
339 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_culture")} {normalizedCulture}");
340 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_wrap_mode")} {(runtime.NoWrap ? TextCatalog.Get(runtime.Language, "wrap_clipping") : TextCatalog.Get(runtime.Language, "wrap_enabled"))}");
341 ConsoleUi.WriteEmptyLine();
342
343 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "loading_configuration"));
344 FIREConfigration config;
345 try
346 {
347 config = FIREConfigration.Load(settings.ConfigPath!);
348 config.EnsureSupportedConfigurationVersion();
349 }
350 catch (Exception ex)
351 {
352 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
353 Console.WriteLine(ex.StackTrace);
354 return 1;
355 }
356
357 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "configuration_loaded"), config.ConfigurationVersion));
358 ConsoleUi.WriteEmptyLine();
359
360 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
361 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_database")} {dbPath}");
362 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "initializing_catalog"));
363
364 var database = new FIREDatabase(dbPath);
365 var catalog = new FIRECatalog(config, database)
366 {
367 Culture = CultureInfo.CurrentUICulture
368 };
369
370 try
371 {
372 catalog.ProgressChanged += (_, progressEvent) =>
373 {
374 AppLifetime.ThrowIfCancellationRequested();
375
376 if (progressEvent.Level == FIRECatalogMessageLevel.Warning)
377 {
378 ConsoleUi.EndProgressLine();
379 ConsoleUi.WriteWarning(runtime, progressEvent.Message);
380 return;
381 }
382
383 if (progressEvent.Level == FIRECatalogMessageLevel.Trace &&
384 !string.IsNullOrWhiteSpace(progressEvent.CurrentFilePath))
385 {
386 ConsoleUi.WriteProgress(
387 runtime,
388 $"[{progressEvent.ProcessedCount}] {TextCatalog.Get(runtime.Language, progressVerbKey)} {progressEvent.CurrentFilePath}");
389 }
390 };
391
392 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "catalog_initialized"));
393 ConsoleUi.WriteEmptyLine();
394
395 beforeAction(catalog);
396 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, progressLineKey));
397
398 var stopwatch = Stopwatch.StartNew();
399 action(catalog, _ => { });
400 stopwatch.Stop();
401
402 ConsoleUi.EndProgressLine();
403 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, summaryKey), stopwatch.Elapsed.TotalSeconds));
404 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "summary_total_records"), database.Count));
405 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "summary_warnings"));
406 ConsoleUi.WriteEmptyLine();
407 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, summaryHintKey));
408 return 0;
409 }
410 catch (OperationCanceledException)
411 {
412 catalog.LogCancelled();
413 ConsoleUi.EndProgressLine();
414 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "operation_canceled"));
415 return 2;
416 }
417 catch (Exception ex)
418 {
419 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
420 Console.WriteLine(ex.StackTrace);
421 return 1;
422 }
423 finally
424 {
425 catalog.Dispose();
426 database.Dispose();
427 }
428 }

430 private static bool TryCopyToClipboard(string text)
431 {
432 if (string.IsNullOrWhiteSpace(text) || !OperatingSystem.IsWindows())
433 {
434 return false;
435 }
436
437 try
438 {
439 var startInfo = new ProcessStartInfo
440 {
441 FileName = "clip",
442 RedirectStandardInput = true,
443 UseShellExecute = false,
444 CreateNoWindow = true
445 };
446
447 using var process = Process.Start(startInfo);
448 if (process is null)
449 {
450 return false;
451 }
452
453 process.StandardInput.Write(text);
454 process.StandardInput.Close();
455 process.WaitForExit(2000);
456 return process.ExitCode == 0;
457 }
458 catch
459 {
460 return false;
461 }
462 }

487internal static class CultureResolver
488{
489 private static readonly Dictionary<string, string> CultureAliases = new(StringComparer.OrdinalIgnoreCase)
490 {
491 ["en"] = "en-US",
492 ["en-en"] = "en-US",
493 ["en-gb"] = "en-GB",
494 ["de"] = "de-DE",
495 ["fr"] = "fr-FR",
496 ["fil"] = "fil-PH",
497 ["tl"] = "fil-PH",
498 ["ph"] = "fil-PH"
499 };
500
501 public static string? ReadCultureArgument(string[] args)
502 {
503 for (var index = 0; index < args.Length; index++)
504 {
505 if ((args[index].Equals("--culture", StringComparison.OrdinalIgnoreCase) || args[index].Equals("-l", StringComparison.OrdinalIgnoreCase))
506 && index + 1 < args.Length)
507 {
508 return args[index + 1];
509 }
510 }
511
512 return null;
513 }
514
515 public static string ResolveCultureCode(string rawCulture)
516 {
517 if (string.IsNullOrWhiteSpace(rawCulture))
518 {
519 return "en-US";
520 }
521
522 var trimmed = rawCulture.Trim();
523 if (CultureAliases.TryGetValue(trimmed, out var alias))
524 {
525 return alias;
526 }
527
528 try
529 {
530 var culture = CultureInfo.GetCultureInfo(trimmed);
531 if (CultureAliases.TryGetValue(culture.Name, out var mappedSpecificCulture))
532 {
533 return mappedSpecificCulture;
534 }
535
536 return culture.Name;
537 }
538 catch
539 {
540 return "en-US";
541 }
542 }
543
544 public static UiLanguage ResolveLanguage(string cultureCode)
545 {
546 if (cultureCode.StartsWith("de", StringComparison.OrdinalIgnoreCase))
547 {
548 return UiLanguage.German;
549 }
550
551 if (cultureCode.StartsWith("fr", StringComparison.OrdinalIgnoreCase))
552 {
553 return UiLanguage.French;
554 }
555
556 if (cultureCode.StartsWith("fil", StringComparison.OrdinalIgnoreCase) || cultureCode.StartsWith("tl", StringComparison.OrdinalIgnoreCase))
557 {
558 return UiLanguage.Filipino;
559 }
560
561 return UiLanguage.English;
562 }
563}

643 private static string FormatLine(string text, bool noWrap)
644 {
645 if (!noWrap)
646 {
647 return text;
648 }
649
650 var width = GetSafeWidth();
651 if (width < 4 || text.Length <= width)
652 {
653 return text;
654 }
655
656 return string.Concat(text.AsSpan(0, width - 3), "...");
657 }
658
659 private static int GetSafeWidth()
660 {
661 try
662 {
663 return Math.Max(Console.WindowWidth - 1, 20);
664 }
665 catch
666 {
667 return 120;
668 }
669 }
