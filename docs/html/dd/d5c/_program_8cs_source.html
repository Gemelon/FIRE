12internal static class ProgramHost
13{
14 public static int Run(string[] args)
15 {
16 var bootstrapCulture = CultureResolver.ResolveCultureCode(CultureResolver.ReadCultureArgument(args) ?? "en-US");
17 var bootstrapLanguage = CultureResolver.ResolveLanguage(bootstrapCulture);
18
19 AppLifetime.CurrentLanguage = bootstrapLanguage;
20
21 Console.CancelKeyPress += (_, eventArgs) =>
22 {
23 eventArgs.Cancel = true;
24 AppLifetime.IsCancellationRequested = true;
25 ConsoleUi.WriteWarning(AppLifetime.CurrentLanguage, AppLifetime.NoWrap, TextCatalog.Get(AppLifetime.CurrentLanguage, "cancel_requested"));
26 };
27
28 var app = new CommandApp();
29 app.Configure(configuration =>
30 {
31 configuration.SetApplicationName("FIRE.Console");
32 configuration.ValidateExamples();
33
34 configuration
35 .AddCommand<CollectCommand>("collect")
36 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_collect"))
37 .WithExample("collect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "en-US")
38 .WithExample("collect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "de-DE", "--clear-database")
39 .WithExample("collect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "fr-FR", "--no-wrap");
40
41 configuration
42 .AddCommand<GenerateCommand>("generate")
43 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_generate"))
44 .WithExample("generate", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "en-US");
45
46 configuration
47 .AddCommand<ExecuteCommand>("execute")
48 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_execute"))
49 .WithExample("execute", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "de-DE");
50
51 configuration
52 .AddCommand<InspectCommand>("inspect")
53 .WithDescription(TextCatalog.Get(bootstrapLanguage, "cmd_inspect"))
54 .WithExample("inspect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "en-US", "--file", "image.jpg")
55 .WithExample("inspect", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "fil-PH", "--file", "image.jpg", "--copy-path");
56
57 configuration
58 .AddCommand<DiagnoseCommand>("diagnose")
59 .WithDescription("Diagnoses path generation for a specific source file and writes a detailed Markdown report.")
60 .WithExample("diagnose", "--config", "ConfigurationFiles\\Configuration.yaml", "--source-path", "D:\\Photos\\image.jpg")
61 .WithExample("diagnose", "--config", "ConfigurationFiles\\Configuration.yaml", "--culture", "de-DE", "--source-path", "D:\\Videos\\video.mp4");
62 });
63
64 try
65 {
66 return app.Run(args);
67 }
68 catch (Exception ex)
69 {
70 ConsoleUi.WriteError(bootstrapLanguage, false, $"{TextCatalog.Get(bootstrapLanguage, "error_prefix")} {ex.Message}");
71 Console.WriteLine(ex.StackTrace);
72 return 1;
73 }
74 }

222 public static int ExecuteInspect(InspectSettings settings, RuntimeContext runtime)
223 {
224 try
225 {
226 if (string.IsNullOrWhiteSpace(settings.ConfigPath))
227 {
228 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_config_required"));
229 return 1;
230 }
231
232 if (string.IsNullOrWhiteSpace(settings.FilePath))
233 {
234 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_file_required"));
235 return 1;
236 }
237
238 var filePath = settings.FilePath;
239 if (!File.Exists(filePath))
240 {
241 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_file_not_found")} {filePath}");
242 return 1;
243 }
244
245 var normalizedCulture = runtime.CultureCode;
246 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = CultureInfo.GetCultureInfo(normalizedCulture);
247
248 ConsoleUi.WriteTitle(runtime, TextCatalog.Get(runtime.Language, "title_inspect"));
249 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_config")} {settings.ConfigPath}");
250 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_culture")} {normalizedCulture}");
251 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_file")} {filePath}");
252 if (!string.IsNullOrWhiteSpace(settings.OutputPath))
253 {
254 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_output")} {settings.OutputPath}");
255 }
256
257 ConsoleUi.WriteEmptyLine();
258 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "loading_configuration"));
259 var config = FIREConfigration.Load(settings.ConfigPath!);
260 config.EnsureSupportedConfigurationVersion();
261 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "configuration_loaded"), config.ConfigurationVersion));
262 ConsoleUi.WriteEmptyLine();
263
264 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
265 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_database")} {dbPath}");
266 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "initializing_catalog"));
267
268 using var database = new FIREDatabase(dbPath);
269 using var catalog = new FIRECatalog(config, database)
270 {
271 Culture = CultureInfo.CurrentUICulture
272 };
273
274 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "catalog_initialized"));
275 ConsoleUi.WriteEmptyLine();
276
277 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "extracting_metadata"));
278 var stopwatch = Stopwatch.StartNew();
279 var metadata = catalog.GetAllAvailableMetadata(filePath);
280 stopwatch.Stop();
281
282 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "metadata_extracted_ms"), stopwatch.Elapsed.TotalMilliseconds));
283 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "metadata_entries_found"), metadata.Count));
284 ConsoleUi.WriteEmptyLine();
285
286 if (metadata.Count > 0)
287 {
288 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "metadata_by_source"));
289 foreach (var sourceGroup in metadata.GroupBy(entry => entry.Source).OrderBy(group => group.Key))
290 {
291 ConsoleUi.WriteLine(runtime, $" - {sourceGroup.Key}: {sourceGroup.Count()}");
292 }
293 ConsoleUi.WriteEmptyLine();
294 }
295 else
296 {
297 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "warn_no_metadata"));
298 ConsoleUi.WriteEmptyLine();
299 }
300
301 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "writing_markdown"));
302 catalog.WriteMetadataToMarkdown(filePath, settings.OutputPath);
303
304 var finalOutputPath = settings.OutputPath ?? Path.Combine(
305 Path.GetDirectoryName(filePath) ?? ".",
306 Path.GetFileNameWithoutExtension(filePath) + ".md");
307 finalOutputPath = Path.GetFullPath(finalOutputPath);
308
309 ConsoleUi.WriteSuccess(runtime, $"{TextCatalog.Get(runtime.Language, "markdown_created")} {finalOutputPath}");
310
311 if (settings.CopyPath)
312 {
313 if (TryCopyToClipboard(finalOutputPath))
314 {
315 ConsoleUi.WriteInfo(runtime, TextCatalog.Get(runtime.Language, "markdown_copied"));
316 }
317 else
318 {
319 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "warn_copy_failed"));
320 }
321 }
322
323 ConsoleUi.WriteEmptyLine();
324 return 0;
325 }
326 catch (OperationCanceledException)
327 {
328 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "operation_canceled"));
329 return 2;
330 }
331 catch (Exception ex)
332 {
333 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
334 Console.WriteLine(ex.StackTrace);
335 return 1;
336 }
337 }

339 public static int ExecuteDiagnose(DiagnoseSettings settings, RuntimeContext runtime)
340 {
341 try
342 {
343 if (string.IsNullOrWhiteSpace(settings.ConfigPath))
344 {
345 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_config_required"));
346 return 1;
347 }
348
349 if (string.IsNullOrWhiteSpace(settings.SourcePath))
350 {
351 ConsoleUi.WriteError(runtime, "Source path is required. Use --source-path to specify the file.");
352 return 1;
353 }
354
355 var sourcePath = settings.SourcePath;
356 if (!File.Exists(sourcePath))
357 {
358 ConsoleUi.WriteError(runtime, $"Source file not found: {sourcePath}");
359 return 1;
360 }
361
362 var normalizedCulture = runtime.CultureCode;
363 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = CultureInfo.GetCultureInfo(normalizedCulture);
364
365 ConsoleUi.WriteTitle(runtime, "FIRE Path Generation Diagnosis");
366 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_config")} {settings.ConfigPath}");
367 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_culture")} {normalizedCulture}");
368 ConsoleUi.WriteLine(runtime, $"Source File: {sourcePath}");
369 ConsoleUi.WriteEmptyLine();
370
371 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "loading_configuration"));
372 var config = FIREConfigration.Load(settings.ConfigPath!);
373 config.EnsureSupportedConfigurationVersion();
374 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "configuration_loaded"), config.ConfigurationVersion));
375 ConsoleUi.WriteEmptyLine();
376
377 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
378 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_database")} {dbPath}");
379
380 if (!File.Exists(dbPath))
381 {
382 ConsoleUi.WriteError(runtime, $"Database not found: {dbPath}");
383 ConsoleUi.WriteWarning(runtime, "Please run 'collect' first to populate the database.");
384 return 1;
385 }
386
387 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "initializing_catalog"));
388
389 using var database = new FIREDatabase(dbPath);
390 using var catalog = new FIRECatalog(config, database)
391 {
392 Culture = CultureInfo.CurrentUICulture
393 };
394
395 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "catalog_initialized"));
396 ConsoleUi.WriteEmptyLine();
397
398 ConsoleUi.WriteLine(runtime, "Generating diagnostic report...");
399 var stopwatch = Stopwatch.StartNew();
400 var reportPath = catalog.DiagnoseGeneration(sourcePath);
401 stopwatch.Stop();
402
403 if (reportPath == null)
404 {
405 ConsoleUi.WriteWarning(runtime, $"Source file not found in database: {sourcePath}");
406 ConsoleUi.WriteWarning(runtime, "This file has not been collected yet. Run 'collect' first.");
407 ConsoleUi.WriteEmptyLine();
408 return 1;
409 }
410
411 ConsoleUi.WriteLine(runtime, $"Diagnosis completed in {stopwatch.Elapsed.TotalMilliseconds:F0} ms");
412 ConsoleUi.WriteEmptyLine();
413 ConsoleUi.WriteSuccess(runtime, $"Diagnostic report written to: {reportPath}");
414 ConsoleUi.WriteEmptyLine();
415
416 return 0;
417 }
418 catch (Exception ex)
419 {
420 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
421 Console.WriteLine(ex.StackTrace);
422 return 1;
423 }
424 }

426 private static int ExecuteCatalogPipeline(
427 CommonCommandSettings settings,
428 RuntimeContext runtime,
429 string titleKey,
430 string progressLineKey,
431 string progressVerbKey,
432 string summaryKey,
433 string summaryHintKey,
434 Action<FIRECatalog> beforeAction,
435 Action<FIRECatalog, Action<string>> action)
436 {
437 if (string.IsNullOrWhiteSpace(settings.ConfigPath))
438 {
439 ConsoleUi.WriteError(runtime, TextCatalog.Get(runtime.Language, "error_config_required"));
440 return 1;
441 }
442
443 var normalizedCulture = runtime.CultureCode;
444 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = CultureInfo.GetCultureInfo(normalizedCulture);
445
446 ConsoleUi.WriteTitle(runtime, TextCatalog.Get(runtime.Language, titleKey));
447 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_config")} {settings.ConfigPath}");
448 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_culture")} {normalizedCulture}");
449 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_wrap_mode")} {(runtime.NoWrap ? TextCatalog.Get(runtime.Language, "wrap_clipping") : TextCatalog.Get(runtime.Language, "wrap_enabled"))}");
450 ConsoleUi.WriteEmptyLine();
451
452 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "loading_configuration"));
453 FIREConfigration config;
454 try
455 {
456 config = FIREConfigration.Load(settings.ConfigPath!);
457 config.EnsureSupportedConfigurationVersion();
458 }
459 catch (Exception ex)
460 {
461 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
462 Console.WriteLine(ex.StackTrace);
463 return 1;
464 }
465
466 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "configuration_loaded"), config.ConfigurationVersion));
467 ConsoleUi.WriteEmptyLine();
468
469 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
470 ConsoleUi.WriteLine(runtime, $"{TextCatalog.Get(runtime.Language, "label_database")} {dbPath}");
471 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "initializing_catalog"));
472
473 var database = new FIREDatabase(dbPath);
474 var catalog = new FIRECatalog(config, database)
475 {
476 Culture = CultureInfo.CurrentUICulture
477 };
478
479 try
480 {
481 catalog.ProgressChanged += (_, progressEvent) =>
482 {
483 AppLifetime.ThrowIfCancellationRequested();
484
485 if (progressEvent.Level == FIRECatalogMessageLevel.Warning)
486 {
487 ConsoleUi.EndProgressLine();
488 ConsoleUi.WriteWarning(runtime, progressEvent.Message);
489 return;
490 }
491
492 if (progressEvent.Level == FIRECatalogMessageLevel.Trace &&
493 !string.IsNullOrWhiteSpace(progressEvent.CurrentFilePath))
494 {
495 ConsoleUi.WriteProgress(
496 runtime,
497 $"[{progressEvent.ProcessedCount}] {TextCatalog.Get(runtime.Language, progressVerbKey)} {progressEvent.CurrentFilePath}");
498 }
499 };
500
501 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "catalog_initialized"));
502 ConsoleUi.WriteEmptyLine();
503
504 beforeAction(catalog);
505 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, progressLineKey));
506
507 var stopwatch = Stopwatch.StartNew();
508 action(catalog, _ => { });
509 stopwatch.Stop();
510
511 ConsoleUi.EndProgressLine();
512 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, summaryKey), stopwatch.Elapsed.TotalSeconds));
513 ConsoleUi.WriteLine(runtime, string.Format(CultureInfo.CurrentCulture, TextCatalog.Get(runtime.Language, "summary_total_records"), database.Count));
514 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, "summary_warnings"));
515 ConsoleUi.WriteEmptyLine();
516 ConsoleUi.WriteLine(runtime, TextCatalog.Get(runtime.Language, summaryHintKey));
517 return 0;
518 }
519 catch (OperationCanceledException)
520 {
521 catalog.LogCancelled();
522 ConsoleUi.EndProgressLine();
523 ConsoleUi.WriteWarning(runtime, TextCatalog.Get(runtime.Language, "operation_canceled"));
524 return 2;
525 }
526 catch (Exception ex)
527 {
528 ConsoleUi.WriteError(runtime, $"{TextCatalog.Get(runtime.Language, "error_prefix")} {ex.Message}");
529 Console.WriteLine(ex.StackTrace);
530 return 1;
531 }
532 finally
533 {
534 catalog.Dispose();
535 database.Dispose();
536 }
537 }

539 private static bool TryCopyToClipboard(string text)
540 {
541 if (string.IsNullOrWhiteSpace(text) || !OperatingSystem.IsWindows())
542 {
543 return false;
544 }
545
546 try
547 {
548 var startInfo = new ProcessStartInfo
549 {
550 FileName = "clip",
551 RedirectStandardInput = true,
552 UseShellExecute = false,
553 CreateNoWindow = true
554 };
555
556 using var process = Process.Start(startInfo);
557 if (process is null)
558 {
559 return false;
560 }
561
562 process.StandardInput.Write(text);
563 process.StandardInput.Close();
564 process.WaitForExit(2000);
565 return process.ExitCode == 0;
566 }
567 catch
568 {
569 return false;
570 }
571 }

588internal enum UiLanguage
589{
590 English,
591 German,
592 French,
593 Filipino
594}

596internal static class CultureResolver
597{
598 private static readonly Dictionary<string, string> CultureAliases = new(StringComparer.OrdinalIgnoreCase)
599 {
600 ["en"] = "en-US",
601 ["en-en"] = "en-US",
602 ["en-gb"] = "en-GB",
603 ["de"] = "de-DE",
604 ["fr"] = "fr-FR",
605 ["fil"] = "fil-PH",
606 ["tl"] = "fil-PH",
607 ["ph"] = "fil-PH"
608 };
609
610 public static string? ReadCultureArgument(string[] args)
611 {
612 for (var index = 0; index < args.Length; index++)
613 {
614 if ((args[index].Equals("--culture", StringComparison.OrdinalIgnoreCase) || args[index].Equals("-l", StringComparison.OrdinalIgnoreCase))
615 && index + 1 < args.Length)
616 {
617 return args[index + 1];
618 }
619 }
620
621 return null;
622 }
623
624 public static string ResolveCultureCode(string rawCulture)
625 {
626 if (string.IsNullOrWhiteSpace(rawCulture))
627 {
628 return "en-US";
629 }
630
631 var trimmed = rawCulture.Trim();
632 if (CultureAliases.TryGetValue(trimmed, out var alias))
633 {
634 return alias;
635 }
636
637 try
638 {
639 var culture = CultureInfo.GetCultureInfo(trimmed);
640 if (CultureAliases.TryGetValue(culture.Name, out var mappedSpecificCulture))
641 {
642 return mappedSpecificCulture;
643 }
644
645 return culture.Name;
646 }
647 catch
648 {
649 return "en-US";
650 }
651 }
652
653 public static UiLanguage ResolveLanguage(string cultureCode)
654 {
655 if (cultureCode.StartsWith("de", StringComparison.OrdinalIgnoreCase))
656 {
657 return UiLanguage.German;
658 }
659
660 if (cultureCode.StartsWith("fr", StringComparison.OrdinalIgnoreCase))
661 {
662 return UiLanguage.French;
663 }
664
665 if (cultureCode.StartsWith("fil", StringComparison.OrdinalIgnoreCase) || cultureCode.StartsWith("tl", StringComparison.OrdinalIgnoreCase))
666 {
667 return UiLanguage.Filipino;
668 }
669
670 return UiLanguage.English;
671 }
672}

752 private static string FormatLine(string text, bool noWrap)
753 {
754 if (!noWrap)
755 {
756 return text;
757 }
758
759 var width = GetSafeWidth();
760 if (width < 4 || text.Length <= width)

762 return text;
763 }
764
765 return string.Concat(text.AsSpan(0, width - 3), "...");
766 }
