7// Simple command-line argument parsing
8if (args.Length == 0)
9{
10 ShowHelp();
11 return 1;
12}
13
14var command = args[0].ToLowerInvariant();
15string? configPath = null;
16string? culture = null;
17
18// Parse arguments
19for (int i = 1; i < args.Length; i++)
20{
21 if ((args[i] == "--config" || args[i] == "-c") && i + 1 < args.Length)
22 {
23 configPath = args[++i];
24 }
25 else if ((args[i] == "--culture" || args[i] == "-l") && i + 1 < args.Length)
26 {
27 culture = args[++i];
28 }
29}
30
31// Validate required arguments
32if (string.IsNullOrWhiteSpace(configPath))
33{
34 Console.WriteLine("ERROR: --config parameter is required");
35 ShowHelp();
36 return 1;
37}
38
39if (string.IsNullOrWhiteSpace(culture))
40{
41 Console.WriteLine("ERROR: --culture parameter is required");
42 ShowHelp();
43 return 1;
44}
45
46// Execute command
47return command switch
48{
49 "collect" => ExecuteCollect(configPath, culture),
50 "generate" => ExecuteGenerate(configPath, culture),
51 "execute" => ExecuteOperations(configPath, culture),
52 _ => ShowHelp()
53};

79static int ExecuteCollect(string configPath, string culture)
80{
81 try
82 {
83 Console.WriteLine($"=== FIRE Collect Files ===");
84 Console.WriteLine($"Config: {configPath}");
85 Console.WriteLine($"Culture: {culture}");
86 Console.WriteLine();
87
88 // Set culture
89 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = new CultureInfo(culture);
90
91 // Load configuration
92 Console.WriteLine("Loading configuration...");
93 var config = FIREConfigration.Load(configPath);
94 config.EnsureSupportedConfigurationVersion();
95 Console.WriteLine($"Configuration loaded successfully (Version: {config.ConfigurationVersion})");
96 Console.WriteLine();
97
98 // Create database path
99 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
100 Console.WriteLine($"Database: {dbPath}");
101
102 // Create catalog
103 Console.WriteLine("Initializing catalog...");
104 using var database = new FIREDatabase(dbPath);
105 using var catalog = new FIRECatalog(config, database);
106 Console.WriteLine("Catalog initialized.");
107 Console.WriteLine();
108
109 // Collect files
110 Console.WriteLine("Collecting files...");
111 var stopwatch = System.Diagnostics.Stopwatch.StartNew();
112 int fileCount = 0;
113 catalog.CollectFiles(filePath =>
114 {
115 fileCount++;
116 Console.Write($"\r[{fileCount}] Processing: {TruncatePath(filePath, 80)}".PadRight(Console.WindowWidth - 1));
117 });
118 stopwatch.Stop();
119 Console.WriteLine();
120 Console.WriteLine($"File collection completed in {stopwatch.Elapsed.TotalSeconds:F2} seconds.");
121 Console.WriteLine($"Total records: {database.Count}");
122 Console.WriteLine("Warnings: missing metadata entries are reported inline with [WARN].");
123 Console.WriteLine();
124
125 Console.WriteLine("Use the database to inspect collected files.");
126 return 0;
127 }
128 catch (Exception ex)
129 {
130 Console.ForegroundColor = ConsoleColor.Red;
131 Console.WriteLine($"ERROR: {ex.Message}");
132 Console.ResetColor();
133 Console.WriteLine(ex.StackTrace);
134 return 1;
135 }
136}

259static string TruncatePath(string path, int maxLength)
260{
261 if (string.IsNullOrEmpty(path) || path.Length <= maxLength)
262 return path;
263
264 // Try to show the end of the path which is usually more relevant
265 return "..." + path.Substring(path.Length - maxLength + 3);
266}
