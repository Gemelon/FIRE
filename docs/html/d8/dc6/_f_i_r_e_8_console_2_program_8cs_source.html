7// Simple command-line argument parsing
8if (args.Length == 0)
9{
10 ShowHelp();
11 return 1;
12}
13
14var command = args[0].ToLowerInvariant();
15string? configPath = null;
16string? culture = null;
17
18// Parse arguments
19for (int i = 1; i < args.Length; i++)
20{
21 if ((args[i] == "--config" || args[i] == "-c") && i + 1 < args.Length)
22 {
23 configPath = args[++i];
24 }
25 else if ((args[i] == "--culture" || args[i] == "-l") && i + 1 < args.Length)
26 {
27 culture = args[++i];
28 }
29}
30
31// Validate required arguments
32if (string.IsNullOrWhiteSpace(configPath))
33{
34 Console.WriteLine("ERROR: --config parameter is required");
35 ShowHelp();
36 return 1;
37}
38
39if (string.IsNullOrWhiteSpace(culture))
40{
41 Console.WriteLine("ERROR: --culture parameter is required");
42 ShowHelp();
43 return 1;
44}
45
46// Execute command
47return command switch
48{
49 "collect" => ExecuteCollect(configPath, culture),
50 "generate" => ExecuteGenerate(configPath, culture),
51 "execute" => ExecuteOperations(configPath, culture),
52 "inspect" => ExecuteInspect(args, configPath, culture),
53 _ => ShowHelp()
54};

85static int ExecuteCollect(string configPath, string culture)
86{
87 try
88 {
89 Console.WriteLine($"=== FIRE Collect Files ===");
90 Console.WriteLine($"Config: {configPath}");
91 Console.WriteLine($"Culture: {culture}");
92 Console.WriteLine();
93
94 // Set culture
95 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = new CultureInfo(culture);
96
97 // Load configuration
98 Console.WriteLine("Loading configuration...");
99 var config = FIREConfigration.Load(configPath);
100 config.EnsureSupportedConfigurationVersion();
101 Console.WriteLine($"Configuration loaded successfully (Version: {config.ConfigurationVersion})");
102 Console.WriteLine();
103
104 // Create database path
105 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
106 Console.WriteLine($"Database: {dbPath}");
107
108 // Create catalog
109 Console.WriteLine("Initializing catalog...");
110 using var database = new FIREDatabase(dbPath);
111 using var catalog = new FIRECatalog(config, database);
112 Console.WriteLine("Catalog initialized.");
113 Console.WriteLine();
114
115 // Collect files
116 Console.WriteLine("Collecting files...");
117 var stopwatch = System.Diagnostics.Stopwatch.StartNew();
118 int fileCount = 0;
119 catalog.CollectFiles(filePath =>
120 {
121 fileCount++;
122 Console.Write($"\r[{fileCount}] Processing: {TruncatePath(filePath, 80)}".PadRight(Console.WindowWidth - 1));
123 });
124 stopwatch.Stop();
125 Console.WriteLine();
126 Console.WriteLine($"File collection completed in {stopwatch.Elapsed.TotalSeconds:F2} seconds.");
127 Console.WriteLine($"Total records: {database.Count}");
128 Console.WriteLine("Warnings: missing metadata entries are reported inline with [WARN].");
129 Console.WriteLine();
130
131 Console.WriteLine("Use the database to inspect collected files.");
132 return 0;
133 }
134 catch (Exception ex)
135 {
136 Console.ForegroundColor = ConsoleColor.Red;
137 Console.WriteLine($"ERROR: {ex.Message}");
138 Console.ResetColor();
139 Console.WriteLine(ex.StackTrace);
140 return 1;
141 }
142}

262static int ExecuteInspect(string[] args, string configPath, string culture)
263{
264 try
265 {
266 // Parse additional arguments for inspect command
267 string? filePath = null;
268 string? outputPath = null;
269
270 for (int i = 1; i < args.Length; i++)
271 {
272 if ((args[i] == "--file" || args[i] == "-f") && i + 1 < args.Length)
273 {
274 filePath = args[++i];
275 }
276 else if ((args[i] == "--output" || args[i] == "-o") && i + 1 < args.Length)
277 {
278 outputPath = args[++i];
279 }
280 }
281
282 if (string.IsNullOrWhiteSpace(filePath))
283 {
284 Console.ForegroundColor = ConsoleColor.Red;
285 Console.WriteLine("ERROR: --file parameter is required for inspect command");
286 Console.ResetColor();
287 Console.WriteLine();
288 ShowHelp();
289 return 1;
290 }
291
292 if (!File.Exists(filePath))
293 {
294 Console.ForegroundColor = ConsoleColor.Red;
295 Console.WriteLine($"ERROR: File not found: {filePath}");
296 Console.ResetColor();
297 return 1;
298 }
299
300 Console.WriteLine($"=== FIRE Inspect File Metadata ===");
301 Console.WriteLine($"Config: {configPath}");
302 Console.WriteLine($"Culture: {culture}");
303 Console.WriteLine($"File: {filePath}");
304 if (!string.IsNullOrWhiteSpace(outputPath))
305 {
306 Console.WriteLine($"Output: {outputPath}");
307 }
308 Console.WriteLine();
309
310 // Set culture
311 CultureInfo.CurrentCulture = CultureInfo.CurrentUICulture = new CultureInfo(culture);
312
313 // Load configuration
314 Console.WriteLine("Loading configuration...");
315 var config = FIREConfigration.Load(configPath);
316 config.EnsureSupportedConfigurationVersion();
317 Console.WriteLine($"Configuration loaded successfully (Version: {config.ConfigurationVersion})");
318 Console.WriteLine();
319
320 // Create database path (temporary, needed for catalog initialization)
321 var dbPath = Path.Combine(config.DataBasePath ?? ".", config.DataBaseFileName ?? "FIRE.db");
322
323 // Create catalog
324 Console.WriteLine("Initializing catalog...");
325 using var database = new FIREDatabase(dbPath);
326 using var catalog = new FIRECatalog(config, database);
327 Console.WriteLine("Catalog initialized.");
328 Console.WriteLine();
329
330 // Extract and display metadata
331 Console.WriteLine("Extracting metadata...");
332 var stopwatch = System.Diagnostics.Stopwatch.StartNew();
333 var metadata = catalog.GetAllAvailableMetadata(filePath);
334 stopwatch.Stop();
335 Console.WriteLine($"Metadata extracted in {stopwatch.Elapsed.TotalMilliseconds:F2} ms.");
336 Console.WriteLine($"Total entries found: {metadata.Count}");
337 Console.WriteLine();
338
339 // Display metadata summary
340 if (metadata.Count > 0)
341 {
342 var groupedBySource = metadata.GroupBy(m => m.Source).OrderBy(g => g.Key);
343 Console.WriteLine("Metadata by Source:");
344 foreach (var sourceGroup in groupedBySource)
345 {
346 Console.WriteLine($" - {sourceGroup.Key}: {sourceGroup.Count()} entries");
347 }
348 Console.WriteLine();
349 }
350 else
351 {
352 Console.ForegroundColor = ConsoleColor.Yellow;
353 Console.WriteLine("WARNING: No metadata could be extracted from this file.");
354 Console.ResetColor();
355 Console.WriteLine();
356 }
357
358 // Write to Markdown
359 Console.WriteLine("Writing Markdown report...");
360 catalog.WriteMetadataToMarkdown(filePath, outputPath);
361
362 var finalOutputPath = outputPath ?? Path.Combine(
363 Path.GetDirectoryName(filePath) ?? ".",
364 Path.GetFileNameWithoutExtension(filePath) + ".md");
365
366 Console.ForegroundColor = ConsoleColor.Green;
367 Console.WriteLine($"✓ Markdown report created: {finalOutputPath}");
368 Console.ResetColor();
369 Console.WriteLine();
370
371 return 0;
372 }
373 catch (Exception ex)
374 {
375 Console.ForegroundColor = ConsoleColor.Red;
376 Console.WriteLine($"ERROR: {ex.Message}");
377 Console.ResetColor();
378 Console.WriteLine(ex.StackTrace);
379 return 1;
380 }
381}

386static string TruncatePath(string path, int maxLength)
387{
388 if (string.IsNullOrEmpty(path) || path.Length <= maxLength)
389 return path;
390
391 // Try to show the end of the path which is usually more relevant
392 return "..." + path.Substring(path.Length - maxLength + 3);
393}
