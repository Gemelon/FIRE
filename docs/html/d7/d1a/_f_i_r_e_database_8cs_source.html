1069 private void PersistCurrentState(string status, bool valid, string? errorMessage)
1070 {
1071 try
1072 {
1073 using var transaction = _context.Database.BeginTransaction();
1074
1075 var existingRecords = _context.FileRecords.Include(record => record.FileMetaDatas).ToList();
1076 if (existingRecords.Count > 0)
1077 {
1078 _context.FileMetaDataRecords.RemoveRange(existingRecords.SelectMany(record => record.FileMetaDatas));
1079 _context.FileRecords.RemoveRange(existingRecords);
1080 _context.SaveChanges();
1081 }
1082
1083 var existingCounterStates = _context.CounterStates.ToList();
1084 if (existingCounterStates.Count > 0)
1085 {
1086 _context.CounterStates.RemoveRange(existingCounterStates);
1087 _context.SaveChanges();
1088 }
1089
1090 foreach (var record in _records)
1091 {
1092 _context.FileRecords.Add(MapToEntity(record));
1093 }
1094
1095 foreach (var counterState in _counterStates.Values)
1096 {
1097 _context.CounterStates.Add(new FIRECounterStateEntity
1098 {
1099 ScopeKey = counterState.ScopeKey,
1100 LastValue = counterState.LastValue,
1101 UpdatedAt = counterState.UpdatedAt
1102 });
1103 }
1104
1105 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
1106 statusEntity.Status = status;
1107 statusEntity.Valid = valid;
1108 statusEntity.ErrorMessage = errorMessage;
1109 statusEntity.TimeStamp = DateTime.UtcNow;
1110
1111 _context.SaveChanges();
1112 transaction.Commit();
1113 CopyToModel(statusEntity, _statusRecord);
1114 }
1115 catch (Exception ex)
1116 {
1117 try
1118 {
1119 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
1120 statusEntity.Status = "Failed";
1121 statusEntity.Valid = false;
1122 statusEntity.ErrorMessage = ex.Message;
1123 statusEntity.TimeStamp = DateTime.UtcNow;
1124 _context.SaveChanges();
1125 CopyToModel(statusEntity, _statusRecord);
1126 }
1127 catch
1128 {
1129 // If the status cannot be updated, the original exception is preserved.
1130 }

1285 internal void SortRecords(List<string>? fileSorting, string? fileSortingOrder)
1286 {
1287 ThrowIfDisposed();
1288
1289 // Nothing to sort when there are zero/one records or no sorting definition.
1290 if (_records.Count <= 1 || fileSorting is null || fileSorting.Count == 0)
1291 return;
1292
1293 // Resolve global default direction first (used when no field-specific direction is provided).
1294 var globalDirection = ParseGlobalSortDirection(fileSortingOrder);
1295 var criteria = new List<SortCriterion>();
1296
1297 // Parse every configured criterion and preserve the configured order.
1298 foreach (var rawCriterion in fileSorting)
1299 {
1300 if (TryParseSortCriterion(rawCriterion, globalDirection, out var criterion))
1301 {
1302 criteria.Add(criterion);
1303 }
1304 }
1305
1306 // Abort if no valid criterion remained after parsing.
1307 if (criteria.Count == 0)
1308 return;
1309
1310 IOrderedEnumerable<FIREDbRecord>? ordered = null;
1311
1312 // Build a dynamic multi-column sort chain: first criterion => OrderBy, subsequent => ThenBy.
1313 foreach (var criterion in criteria)
1314 {
1315 if (ordered is null)
1316 {
1317 ordered = criterion.Direction == SortDirection.Descending
1318 ? _records.OrderByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1319 : _records.OrderBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1320 }
1321 else
1322 {
1323 ordered = criterion.Direction == SortDirection.Descending
1324 ? ordered.ThenByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1325 : ordered.ThenBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1326 }
1327 }
1328
1329 if (ordered is null)
1330 return;
1331
1332 // Deterministic tie-breakers avoid unstable processing order for identical sort values.
1333 var sorted = ordered
1334 .ThenBy(record => record.SourceFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1335 .ThenBy(record => record.TargetFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1336 .ToList();
1337
1338 // Replace the in-memory sequence so generate/execute consume the new order.
1339 _records.Clear();
1340 _records.AddRange(sorted);
1341 }

1370 private static bool TryParseSortCriterion(string? rawCriterion, SortDirection defaultDirection, out SortCriterion criterion)
1371 {
1372 criterion = default;
1373
1374 if (string.IsNullOrWhiteSpace(rawCriterion))
1375 return false;
1376
1377 var token = rawCriterion.Trim();
1378
1379 // Per-field direction is encoded as <criterion>:<direction>.
1380 var separatorIndex = token.LastIndexOf(':');
1381 if (separatorIndex > 0 && separatorIndex < token.Length - 1)
1382 {
1383 var criterionName = token[..separatorIndex].Trim();
1384 var directionToken = token[(separatorIndex + 1)..].Trim();
1385
1386 if (!string.IsNullOrWhiteSpace(criterionName) && TryParseSortDirection(directionToken, out var parsedDirection))
1387 {
1388 criterion = new SortCriterion(criterionName, parsedDirection);
1389 return true;
1390 }
1391 }
1392
1393 // No valid field-specific direction found => fall back to global direction.
1394 criterion = new SortCriterion(token, defaultDirection);
1395 return true;
1396 }

1404 private static bool TryParseSortDirection(string? value, out SortDirection direction)
1405 {
1406 direction = SortDirection.Ascending;
1407
1408 if (string.IsNullOrWhiteSpace(value))
1409 return false;
1410
1411 switch (value.Trim().ToUpperInvariant())
1412 {
1413 case "ASC":
1414 case "ASCENDING":
1415 case "AUFSTEIGEND":
1416 direction = SortDirection.Ascending;
1417 return true;
1418 case "DESC":
1419 case "DESCENDING":
1420 case "ABSTEIGEND":
1421 direction = SortDirection.Descending;
1422 return true;
1423 default:
1424 return false;
1425 }
1426 }

1449 private sealed class SortValueComparer : IComparer<IComparable?>
1450 {
1451 public static readonly SortValueComparer Instance = new();
1452
1453 public int Compare(IComparable? x, IComparable? y)
1454 {
1455 if (ReferenceEquals(x, y))
1456 return 0;
1457
1458 // Nulls last: records without sort value are processed after records with values.
1459 if (x is null)
1460 return 1;
1461 if (y is null)
1462 return -1;
1463
1464 if (x.GetType() == y.GetType())
1465 return x.CompareTo(y);
1466
1467 var left = Convert.ToString(x, CultureInfo.InvariantCulture) ?? string.Empty;
1468 var right = Convert.ToString(y, CultureInfo.InvariantCulture) ?? string.Empty;
1469 return StringComparer.OrdinalIgnoreCase.Compare(left, right);
1470 }
1471 }

1497 private static IComparable? GetSortValue(FIREDbRecord record, string criterion)
1498 {
1499 var parts = criterion.Split('.', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
1500 if (parts.Length == 0)
1501 return null;
1502
1503 var key = parts[0];
1504 var selector = parts.Length > 1 ? parts[1] : null;
1505
1506 var metadataValue = record.FileMetaDatas.FirstOrDefault(m =>
1507 string.Equals(m.Key, key, StringComparison.OrdinalIgnoreCase))?.Value;
1508
1509 if (string.IsNullOrWhiteSpace(metadataValue))
1510 return null;
1511
1512 if (selector == null)
1513 {
1514 if (DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var dateTimeValue))
1515 return dateTimeValue;
1516
1517 if (decimal.TryParse(metadataValue, NumberStyles.Any, CultureInfo.InvariantCulture, out var numberValue))
1518 return numberValue;
1519
1520 return metadataValue;
1521 }
1522
1523 if (!DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var parsedDateTime))
1524 return metadataValue;
1525
1526 return selector.ToLowerInvariant() switch
1527 {
1528 "year" => parsedDateTime.Year,
1529 "month" => parsedDateTime.Month,
1530 "day" => parsedDateTime.Day,
1531 "hour" => parsedDateTime.Hour,
1532 "minute" => parsedDateTime.Minute,
1533 "second" => parsedDateTime.Second,
1534 _ => metadataValue
1535 };
1536 }
