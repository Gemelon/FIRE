402 public FIREDatabase(string databaseFilePath, bool recreateIfExists = false)
403 {
404 ArgumentException.ThrowIfNullOrWhiteSpace(databaseFilePath);
405
406 _databaseFilePath = Path.GetFullPath(databaseFilePath);
407 var directory = Path.GetDirectoryName(_databaseFilePath);
408 if (!string.IsNullOrWhiteSpace(directory))
409 {
410 Directory.CreateDirectory(directory);
411 }
412
413 var options = new DbContextOptionsBuilder<FIREDatabaseContext>()
414 .UseSqlite($"Data Source={_databaseFilePath}")
415 .Options;
416
417 _context = new FIREDatabaseContext(options);
418
419 if (recreateIfExists && File.Exists(_databaseFilePath))
420 {
421 _context.Database.EnsureDeleted();
422 }
423
424 _context.Database.EnsureCreated();
425 EnsureStatusRowExists();
426 Reload();
427 }

731 private void EnsureStatusRowExists()
732 {
733 if (_context.StatusRecords.Any(x => x.Id == FIREDatabaseStatusEntity.SingletonId))
734 {
735 return;
736 }
737
738 _context.StatusRecords.Add(new FIREStatusEntity
739 {
740 Id = FIREDatabaseStatusEntity.SingletonId,
741 Status = "Initialized",
742 Valid = true,
743 TimeStamp = DateTime.UtcNow,
744 ErrorMessage = null
745 });
746
747 _context.SaveChanges();
748 CopyToModel(_context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId), _statusRecord);
749 }
750
751 private void PersistCurrentState(string status, bool valid, string? errorMessage)
752 {
753 try
754 {
755 using var transaction = _context.Database.BeginTransaction();
756
757 var existingRecords = _context.FileRecords.Include(record => record.FileMetaDatas).ToList();
758 if (existingRecords.Count > 0)
759 {
760 _context.FileMetaDataRecords.RemoveRange(existingRecords.SelectMany(record => record.FileMetaDatas));
761 _context.FileRecords.RemoveRange(existingRecords);
762 _context.SaveChanges();
763 }
764
765 foreach (var record in _records)
766 {
767 _context.FileRecords.Add(MapToEntity(record));
768 }
769
770 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
771 statusEntity.Status = status;
772 statusEntity.Valid = valid;
773 statusEntity.ErrorMessage = errorMessage;
774 statusEntity.TimeStamp = DateTime.UtcNow;
775
776 _context.SaveChanges();
777 transaction.Commit();
778 CopyToModel(statusEntity, _statusRecord);
779 }
780 catch (Exception ex)
781 {
782 try
783 {
784 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
785 statusEntity.Status = "Failed";
786 statusEntity.Valid = false;
787 statusEntity.ErrorMessage = ex.Message;
788 statusEntity.TimeStamp = DateTime.UtcNow;
789 _context.SaveChanges();
790 CopyToModel(statusEntity, _statusRecord);
791 }
792 catch
793 {
794 // If the status cannot be updated, the original exception is preserved.
795 }
796
797 throw;
798 }
799 }

801 private static FIREDbRecord MapToModel(FIREDbRecordEntity entity)
802 {
803 var model = new FIREDbRecord
804 {
805 VolumeSerialNumber = entity.VolumeSerialNumber,
806 FileId = entity.FileId,
807 SourceFilePath = entity.SourceFilePath,
808 TargetFilePath = entity.TargetFilePath,
809 FileMetaDatas = entity.FileMetaDatas
810 .OrderBy(meta => meta.Id)
811 .Select(meta => new FIREFileMetaData
812 {
813 Id = meta.Id,
814 Key = meta.Key,
815 Value = meta.Value,
816 TypeName = meta.TypeName,
817 DataSource = meta.DataSource
818 })
819 .ToList()
820 };
821
822 return model;
823 }

841 Value = meta.Value,
842 TypeName = meta.TypeName,

844 });
