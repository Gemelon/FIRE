469 public FIREDatabase(string databaseFilePath, bool recreateIfExists = false)
470 {
471 ArgumentException.ThrowIfNullOrWhiteSpace(databaseFilePath);
472
473 _databaseFilePath = Path.GetFullPath(databaseFilePath);
474 var directory = Path.GetDirectoryName(_databaseFilePath);
475 if (!string.IsNullOrWhiteSpace(directory))
476 {
477 Directory.CreateDirectory(directory);
478 }
479
480 var options = new DbContextOptionsBuilder<FIREDatabaseContext>()
481 .UseSqlite($"Data Source={_databaseFilePath}")
482 .Options;
483
484 _context = new FIREDatabaseContext(options);
485
486 if (recreateIfExists && File.Exists(_databaseFilePath))
487 {
488 _context.Database.EnsureDeleted();
489 }
490
491 _context.Database.EnsureCreated();
492 EnsureStatusRowExists();
493 Reload();

798 private void EnsureStatusRowExists()
799 {
800 if (_context.StatusRecords.Any(x => x.Id == FIREDatabaseStatusEntity.SingletonId))
801 {
802 return;
803 }
804
805 _context.StatusRecords.Add(new FIREStatusEntity
806 {
807 Id = FIREDatabaseStatusEntity.SingletonId,
808 Status = "Initialized",
809 Valid = true,
810 TimeStamp = DateTime.UtcNow,
811 ErrorMessage = null
812 });
813
814 _context.SaveChanges();
815 CopyToModel(_context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId), _statusRecord);
816 }

818 private void PersistCurrentState(string status, bool valid, string? errorMessage)
819 {
820 try
821 {
822 using var transaction = _context.Database.BeginTransaction();
823
824 var existingRecords = _context.FileRecords.Include(record => record.FileMetaDatas).ToList();
825 if (existingRecords.Count > 0)
826 {
827 _context.FileMetaDataRecords.RemoveRange(existingRecords.SelectMany(record => record.FileMetaDatas));
828 _context.FileRecords.RemoveRange(existingRecords);
829 _context.SaveChanges();
830 }
831
832 foreach (var record in _records)
833 {
834 _context.FileRecords.Add(MapToEntity(record));
835 }
836
837 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
838 statusEntity.Status = status;
839 statusEntity.Valid = valid;
840 statusEntity.ErrorMessage = errorMessage;
841 statusEntity.TimeStamp = DateTime.UtcNow;
842
843 _context.SaveChanges();
844 transaction.Commit();
845 CopyToModel(statusEntity, _statusRecord);
846 }
847 catch (Exception ex)
848 {
849 try
850 {
851 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
852 statusEntity.Status = "Failed";
853 statusEntity.Valid = false;
854 statusEntity.ErrorMessage = ex.Message;
855 statusEntity.TimeStamp = DateTime.UtcNow;
856 _context.SaveChanges();
857 CopyToModel(statusEntity, _statusRecord);
858 }
859 catch
860 {
861 // If the status cannot be updated, the original exception is preserved.
862 }
863
864 throw;
865 }
