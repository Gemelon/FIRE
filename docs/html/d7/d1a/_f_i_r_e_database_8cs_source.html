1198 ScopeKey = counterState.ScopeKey,
1199 LastValue = counterState.LastValue,

1202 }

1204 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
1205 statusEntity.Status = status;

1388 internal void SortRecords(List<string>? fileSorting, string? fileSortingOrder)
1389 {
1390 ThrowIfDisposed();
1391
1392 // Nothing to sort when there are zero/one records or no sorting definition.
1393 if (_records.Count <= 1 || fileSorting is null || fileSorting.Count == 0)
1394 return;
1395
1396 // Resolve global default direction first (used when no field-specific direction is provided).
1397 var globalDirection = ParseGlobalSortDirection(fileSortingOrder);
1398 var criteria = new List<SortCriterion>();
1399
1400 // Parse every configured criterion and preserve the configured order.
1401 foreach (var rawCriterion in fileSorting)
1402 {
1403 if (TryParseSortCriterion(rawCriterion, globalDirection, out var criterion))
1404 {
1405 criteria.Add(criterion);
1406 }
1407 }
1408
1409 // Abort if no valid criterion remained after parsing.
1410 if (criteria.Count == 0)
1411 return;
1412
1413 IOrderedEnumerable<FIREDbRecord>? ordered = null;
1414
1415 // Build a dynamic multi-column sort chain: first criterion => OrderBy, subsequent => ThenBy.
1416 foreach (var criterion in criteria)
1417 {
1418 if (ordered is null)
1419 {
1420 ordered = criterion.Direction == SortDirection.Descending
1421 ? _records.OrderByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1422 : _records.OrderBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1423 }
1424 else
1425 {
1426 ordered = criterion.Direction == SortDirection.Descending
1427 ? ordered.ThenByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1428 : ordered.ThenBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1429 }
1430 }
1431
1432 if (ordered is null)
1433 return;
1434
1435 // Deterministic tie-breakers avoid unstable processing order for identical sort values.
1436 var sorted = ordered
1437 .ThenBy(record => record.SourceFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1438 .ThenBy(record => record.TargetFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1439 .ToList();
1440
1441 // Replace the in-memory sequence so generate/execute consume the new order.
1442 _records.Clear();
1443 _records.AddRange(sorted);
1444 }

1473 private static bool TryParseSortCriterion(string? rawCriterion, SortDirection defaultDirection, out SortCriterion criterion)
1474 {
1475 criterion = default;
1476
1477 if (string.IsNullOrWhiteSpace(rawCriterion))
1478 return false;
1479
1480 var token = rawCriterion.Trim();
1481
1482 // Per-field direction is encoded as <criterion>:<direction>.
1483 var separatorIndex = token.LastIndexOf(':');
1484 if (separatorIndex > 0 && separatorIndex < token.Length - 1)
1485 {
1486 var criterionName = token[..separatorIndex].Trim();
1487 var directionToken = token[(separatorIndex + 1)..].Trim();
1488
1489 if (!string.IsNullOrWhiteSpace(criterionName) && TryParseSortDirection(directionToken, out var parsedDirection))
1490 {
1491 criterion = new SortCriterion(criterionName, parsedDirection);
1492 return true;
1493 }
1494 }
1495
1496 // No valid field-specific direction found => fall back to global direction.
1497 criterion = new SortCriterion(token, defaultDirection);
1498 return true;
1499 }

1507 private static bool TryParseSortDirection(string? value, out SortDirection direction)
1508 {
1509 direction = SortDirection.Ascending;
1510
1511 if (string.IsNullOrWhiteSpace(value))
1512 return false;
1513
1514 switch (value.Trim().ToUpperInvariant())
1515 {
1516 case "ASC":
1517 case "ASCENDING":
1518 case "AUFSTEIGEND":
1519 direction = SortDirection.Ascending;
1520 return true;
1521 case "DESC":
1522 case "DESCENDING":
1523 case "ABSTEIGEND":
1524 direction = SortDirection.Descending;
1525 return true;
1526 default:
1527 return false;
1528 }
1529 }

1552 private sealed class SortValueComparer : IComparer<IComparable?>
1553 {
1554 public static readonly SortValueComparer Instance = new();
1555
1556 public int Compare(IComparable? x, IComparable? y)
1557 {
1558 if (ReferenceEquals(x, y))
1559 return 0;
1560
1561 // Nulls last: records without sort value are processed after records with values.
1562 if (x is null)
1563 return 1;
1564 if (y is null)
1565 return -1;
1566
1567 if (x.GetType() == y.GetType())
1568 return x.CompareTo(y);
1569
1570 var left = Convert.ToString(x, CultureInfo.InvariantCulture) ?? string.Empty;
1571 var right = Convert.ToString(y, CultureInfo.InvariantCulture) ?? string.Empty;
1572 return StringComparer.OrdinalIgnoreCase.Compare(left, right);
1573 }
1574 }

1600 private static IComparable? GetSortValue(FIREDbRecord record, string criterion)
1601 {
1602 var parts = criterion.Split('.', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
1603 if (parts.Length == 0)
1604 return null;
1605
1606 var key = parts[0];
1607 var selector = parts.Length > 1 ? parts[1] : null;
1608
1609 var metadataValue = record.FileMetaDatas.FirstOrDefault(m =>
1610 string.Equals(m.Key, key, StringComparison.OrdinalIgnoreCase))?.Value;
1611
1612 if (string.IsNullOrWhiteSpace(metadataValue))
1613 return null;
1614
1615 if (selector == null)
1616 {
1617 if (DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var dateTimeValue))
1618 return dateTimeValue;
1619
1620 if (decimal.TryParse(metadataValue, NumberStyles.Any, CultureInfo.InvariantCulture, out var numberValue))
1621 return numberValue;
1622
1623 return metadataValue;
1624 }
1625
1626 if (!DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var parsedDateTime))
1627 return metadataValue;
1628
1629 return selector.ToLowerInvariant() switch
1630 {
1631 "year" => parsedDateTime.Year,
1632 "month" => parsedDateTime.Month,
1633 "day" => parsedDateTime.Day,
1634 "hour" => parsedDateTime.Hour,
1635 "minute" => parsedDateTime.Minute,
1636 "second" => parsedDateTime.Second,
1637 _ => metadataValue
1638 };
1639 }
