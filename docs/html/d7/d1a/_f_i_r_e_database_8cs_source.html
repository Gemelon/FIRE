1209 internal void SortRecords(List<string>? fileSorting, string? fileSortingOrder)
1210 {
1211 ThrowIfDisposed();
1212
1213 // Nothing to sort when there are zero/one records or no sorting definition.
1214 if (_records.Count <= 1 || fileSorting is null || fileSorting.Count == 0)
1215 return;
1216
1217 // Resolve global default direction first (used when no field-specific direction is provided).
1218 var globalDirection = ParseGlobalSortDirection(fileSortingOrder);
1219 var criteria = new List<SortCriterion>();
1220
1221 // Parse every configured criterion and preserve the configured order.
1222 foreach (var rawCriterion in fileSorting)
1223 {
1224 if (TryParseSortCriterion(rawCriterion, globalDirection, out var criterion))
1225 {
1226 criteria.Add(criterion);
1227 }
1228 }
1229
1230 // Abort if no valid criterion remained after parsing.
1231 if (criteria.Count == 0)
1232 return;
1233
1234 IOrderedEnumerable<FIREDbRecord>? ordered = null;
1235
1236 // Build a dynamic multi-column sort chain: first criterion => OrderBy, subsequent => ThenBy.
1237 foreach (var criterion in criteria)
1238 {
1239 if (ordered is null)
1240 {
1241 ordered = criterion.Direction == SortDirection.Descending
1242 ? _records.OrderByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1243 : _records.OrderBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1244 }
1245 else
1246 {
1247 ordered = criterion.Direction == SortDirection.Descending
1248 ? ordered.ThenByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1249 : ordered.ThenBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1250 }
1251 }
1252
1253 if (ordered is null)
1254 return;
1255
1256 // Deterministic tie-breakers avoid unstable processing order for identical sort values.
1257 var sorted = ordered
1258 .ThenBy(record => record.SourceFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1259 .ThenBy(record => record.TargetFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1260 .ToList();
1261
1262 // Replace the in-memory sequence so generate/execute consume the new order.
1263 _records.Clear();
1264 _records.AddRange(sorted);
1265 }

1294 private static bool TryParseSortCriterion(string? rawCriterion, SortDirection defaultDirection, out SortCriterion criterion)
1295 {
1296 criterion = default;
1297
1298 if (string.IsNullOrWhiteSpace(rawCriterion))
1299 return false;
1300
1301 var token = rawCriterion.Trim();
1302
1303 // Per-field direction is encoded as <criterion>:<direction>.
1304 var separatorIndex = token.LastIndexOf(':');
1305 if (separatorIndex > 0 && separatorIndex < token.Length - 1)
1306 {
1307 var criterionName = token[..separatorIndex].Trim();
1308 var directionToken = token[(separatorIndex + 1)..].Trim();
1309
1310 if (!string.IsNullOrWhiteSpace(criterionName) && TryParseSortDirection(directionToken, out var parsedDirection))
1311 {
1312 criterion = new SortCriterion(criterionName, parsedDirection);
1313 return true;
1314 }
1315 }
1316
1317 // No valid field-specific direction found => fall back to global direction.
1318 criterion = new SortCriterion(token, defaultDirection);
1319 return true;
1320 }

1328 private static bool TryParseSortDirection(string? value, out SortDirection direction)
1329 {
1330 direction = SortDirection.Ascending;
1331
1332 if (string.IsNullOrWhiteSpace(value))
1333 return false;
1334
1335 switch (value.Trim().ToUpperInvariant())
1336 {
1337 case "ASC":
1338 case "ASCENDING":
1339 case "AUFSTEIGEND":
1340 direction = SortDirection.Ascending;
1341 return true;
1342 case "DESC":
1343 case "DESCENDING":
1344 case "ABSTEIGEND":
1345 direction = SortDirection.Descending;
1346 return true;
1347 default:
1348 return false;
1349 }
1350 }

1373 private sealed class SortValueComparer : IComparer<IComparable?>
1374 {
1375 public static readonly SortValueComparer Instance = new();
1376
1377 public int Compare(IComparable? x, IComparable? y)
1378 {
1379 if (ReferenceEquals(x, y))
1380 return 0;
1381
1382 // Nulls last: records without sort value are processed after records with values.
1383 if (x is null)
1384 return 1;
1385 if (y is null)
1386 return -1;
1387
1388 if (x.GetType() == y.GetType())
1389 return x.CompareTo(y);
1390
1391 var left = Convert.ToString(x, CultureInfo.InvariantCulture) ?? string.Empty;
1392 var right = Convert.ToString(y, CultureInfo.InvariantCulture) ?? string.Empty;
1393 return StringComparer.OrdinalIgnoreCase.Compare(left, right);
1394 }
1395 }

1421 private static IComparable? GetSortValue(FIREDbRecord record, string criterion)
1422 {
1423 var parts = criterion.Split('.', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
1424 if (parts.Length == 0)
1425 return null;
1426
1427 var key = parts[0];
1428 var selector = parts.Length > 1 ? parts[1] : null;
1429
1430 var metadataValue = record.FileMetaDatas.FirstOrDefault(m =>
1431 string.Equals(m.Key, key, StringComparison.OrdinalIgnoreCase))?.Value;
1432
1433 if (string.IsNullOrWhiteSpace(metadataValue))
1434 return null;
1435
1436 if (selector == null)
1437 {
1438 if (DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var dateTimeValue))
1439 return dateTimeValue;
1440
1441 if (decimal.TryParse(metadataValue, NumberStyles.Any, CultureInfo.InvariantCulture, out var numberValue))
1442 return numberValue;
1443
1444 return metadataValue;
1445 }
1446
1447 if (!DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var parsedDateTime))
1448 return metadataValue;
1449
1450 return selector.ToLowerInvariant() switch
1451 {
1452 "year" => parsedDateTime.Year,
1453 "month" => parsedDateTime.Month,
1454 "day" => parsedDateTime.Day,
1455 "hour" => parsedDateTime.Hour,
1456 "minute" => parsedDateTime.Minute,
1457 "second" => parsedDateTime.Second,
1458 _ => metadataValue
1459 };
1460 }
