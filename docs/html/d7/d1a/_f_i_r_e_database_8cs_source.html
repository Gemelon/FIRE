1176 {
1177 try
1178 {
1179 using var transaction = _context.Database.BeginTransaction();
1180
1181 var existingRecords = _context.FileRecords.Include(record => record.FileMetaDatas).ToList();

1183 {
1184 _context.FileMetaDataRecords.RemoveRange(existingRecords.SelectMany(record => record.FileMetaDatas));
1185 _context.FileRecords.RemoveRange(existingRecords);

1188

1190 if (existingCounterStates.Count > 0)

1192 _context.CounterStates.RemoveRange(existingCounterStates);

1194 }
1195
1196 foreach (var record in _records)
1197 {
1198 _context.FileRecords.Add(MapToEntity(record));

1200

1202 {

1204 {
1205 ScopeKey = counterState.ScopeKey,
1206 LastValue = counterState.LastValue,
1207 UpdatedAt = counterState.UpdatedAt
1208 });
1209 }
1210
1211 var statusEntity = _context.StatusRecords.Single(x => x.Id == FIREDatabaseStatusEntity.SingletonId);
1212 statusEntity.Status = status;
1213 statusEntity.Valid = valid;
1214 statusEntity.ErrorMessage = errorMessage;
1215 statusEntity.TimeStamp = DateTime.UtcNow;

1395 internal void SortRecords(List<string>? fileSorting, string? fileSortingOrder)
1396 {
1397 ThrowIfDisposed();
1398
1399 // Nothing to sort when there are zero/one records or no sorting definition.
1400 if (_records.Count <= 1 || fileSorting is null || fileSorting.Count == 0)
1401 return;
1402
1403 // Resolve global default direction first (used when no field-specific direction is provided).
1404 var globalDirection = ParseGlobalSortDirection(fileSortingOrder);
1405 var criteria = new List<SortCriterion>();
1406
1407 // Parse every configured criterion and preserve the configured order.
1408 foreach (var rawCriterion in fileSorting)
1409 {
1410 if (TryParseSortCriterion(rawCriterion, globalDirection, out var criterion))
1411 {
1412 criteria.Add(criterion);
1413 }
1414 }
1415
1416 // Abort if no valid criterion remained after parsing.
1417 if (criteria.Count == 0)
1418 return;
1419
1420 IOrderedEnumerable<FIREDbRecord>? ordered = null;
1421
1422 // Build a dynamic multi-column sort chain: first criterion => OrderBy, subsequent => ThenBy.
1423 foreach (var criterion in criteria)
1424 {
1425 if (ordered is null)
1426 {
1427 ordered = criterion.Direction == SortDirection.Descending
1428 ? _records.OrderByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1429 : _records.OrderBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1430 }
1431 else
1432 {
1433 ordered = criterion.Direction == SortDirection.Descending
1434 ? ordered.ThenByDescending(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance)
1435 : ordered.ThenBy(record => GetSortValue(record, criterion.Name), SortValueComparer.Instance);
1436 }
1437 }
1438
1439 if (ordered is null)
1440 return;
1441
1442 // Deterministic tie-breakers avoid unstable processing order for identical sort values.
1443 var sorted = ordered
1444 .ThenBy(record => record.SourceFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1445 .ThenBy(record => record.TargetFilePath ?? string.Empty, StringComparer.OrdinalIgnoreCase)
1446 .ToList();
1447
1448 // Replace the in-memory sequence so generate/execute consume the new order.
1449 _records.Clear();
1450 _records.AddRange(sorted);
1451 }

1458 private static SortDirection ParseGlobalSortDirection(string? fileSortingOrder)
1459 {

1480 private static bool TryParseSortCriterion(string? rawCriterion, SortDirection defaultDirection, out SortCriterion criterion)
1481 {
1482 criterion = default;
1483
1484 if (string.IsNullOrWhiteSpace(rawCriterion))
1485 return false;
1486
1487 var token = rawCriterion.Trim();
1488
1489 // Per-field direction is encoded as <criterion>:<direction>.
1490 var separatorIndex = token.LastIndexOf(':');
1491 if (separatorIndex > 0 && separatorIndex < token.Length - 1)
1492 {
1493 var criterionName = token[..separatorIndex].Trim();
1494 var directionToken = token[(separatorIndex + 1)..].Trim();
1495
1496 if (!string.IsNullOrWhiteSpace(criterionName) && TryParseSortDirection(directionToken, out var parsedDirection))
1497 {
1498 criterion = new SortCriterion(criterionName, parsedDirection);
1499 return true;
1500 }
1501 }
1502
1503 // No valid field-specific direction found => fall back to global direction.
1504 criterion = new SortCriterion(token, defaultDirection);
1505 return true;
1506 }

1514 private static bool TryParseSortDirection(string? value, out SortDirection direction)
1515 {
1516 direction = SortDirection.Ascending;
1517
1518 if (string.IsNullOrWhiteSpace(value))
1519 return false;
1520
1521 switch (value.Trim().ToUpperInvariant())
1522 {
1523 case "ASC":
1524 case "ASCENDING":
1525 case "AUFSTEIGEND":
1526 direction = SortDirection.Ascending;
1527 return true;
1528 case "DESC":
1529 case "DESCENDING":
1530 case "ABSTEIGEND":
1531 direction = SortDirection.Descending;
1532 return true;
1533 default:
1534 return false;
1535 }
1536 }

1559 private sealed class SortValueComparer : IComparer<IComparable?>
1560 {
1561 public static readonly SortValueComparer Instance = new();
1562
1572 public int Compare(IComparable? x, IComparable? y)
1573 {
1574 if (ReferenceEquals(x, y))
1575 return 0;
1576
1577 // Nulls last: records without sort value are processed after records with values.
1578 if (x is null)
1579 return 1;
1580 if (y is null)
1581 return -1;
1582
1583 if (x.GetType() == y.GetType())
1584 return x.CompareTo(y);
1585
1586 var left = Convert.ToString(x, CultureInfo.InvariantCulture) ?? string.Empty;
1587 var right = Convert.ToString(y, CultureInfo.InvariantCulture) ?? string.Empty;
1588 return StringComparer.OrdinalIgnoreCase.Compare(left, right);
1589 }
1590 }

1616 private static IComparable? GetSortValue(FIREDbRecord record, string criterion)
1617 {
1618 var parts = criterion.Split('.', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
1619 if (parts.Length == 0)
1620 return null;
1621
1622 var key = parts[0];
1623 var selector = parts.Length > 1 ? parts[1] : null;
1624
1625 var metadataValue = record.FileMetaDatas.FirstOrDefault(m =>
1626 string.Equals(m.Key, key, StringComparison.OrdinalIgnoreCase))?.Value;
1627
1628 if (string.IsNullOrWhiteSpace(metadataValue))
1629 return null;
1630
1631 if (selector == null)
1632 {
1633 if (DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var dateTimeValue))
1634 return dateTimeValue;
1635
1636 if (decimal.TryParse(metadataValue, NumberStyles.Any, CultureInfo.InvariantCulture, out var numberValue))
1637 return numberValue;
1638
1639 return metadataValue;
1640 }
1641
1642 if (!DateTime.TryParse(metadataValue, CultureInfo.InvariantCulture, DateTimeStyles.AllowWhiteSpaces, out var parsedDateTime))
1643 return metadataValue;
1644
1645 return selector.ToLowerInvariant() switch
1646 {
1647 "year" => parsedDateTime.Year,
1648 "month" => parsedDateTime.Month,
1649 "day" => parsedDateTime.Day,
1650 "hour" => parsedDateTime.Hour,
1651 "minute" => parsedDateTime.Minute,
1652 "second" => parsedDateTime.Second,
1653 _ => metadataValue
1654 };
1655 }
